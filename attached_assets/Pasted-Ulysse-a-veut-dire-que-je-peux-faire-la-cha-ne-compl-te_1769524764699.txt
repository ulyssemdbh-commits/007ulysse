Je pars sur une logique générique, tu adaptes les hooks vers ton storage / Whisper.

// services/audioTranslateService.ts
import { db } from "../db";
import { ulysseFiles } from "@shared/schema";
import OpenAI from "openai";
import { translationService } from "./translationService";
// à adapter selon ton implémentation Whisper et TTS
import { transcribeAudioFile } from "./whisperService"; 
import { synthesizeSpeech } from "./ttsService";

const openai = new OpenAI({
  apiKey: process.env.AI_INTEGRATIONS_OPENAI_API_KEY,
  baseURL: process.env.AI_INTEGRATIONS_OPENAI_BASE_URL,
});

export interface AudioTranslateRequest {
  userId: number;
  fileId: number;
  targetLang: string;
  sourceLang?: string;         // "auto" par défaut
  domain?: "general" | "sports" | "tech" | "business";
  tone?: "neutral" | "formal" | "casual";
  generateAudio?: boolean;     // default: true
}

export interface AudioTranslateResult {
  success: boolean;
  error?: string;
  sourceLang: string;
  targetLang: string;
  originalTranscript: string;
  translatedTranscript: string;
  audioFileId?: number;        // ID du fichier audio généré
  metadata: {
    durationMs?: number;
    fromCache: boolean;
  };
}

export class AudioTranslateService {
  async translateAudio(req: AudioTranslateRequest): Promise<AudioTranslateResult> {
    const start = Date.now();

    try {
      // 1) Récupérer le fichier
      const [file] = await db
        .select()
        .from(ulysseFiles)
        .where(
          // sécurité: userId + fileId
          ulysseFiles.id.eq(req.fileId).and(ulysseFiles.userId.eq(req.userId as any))
        );

      if (!file) {
        return {
          success: false,
          error: "Fichier introuvable",
          sourceLang: req.sourceLang || "unknown",
          targetLang: req.targetLang,
          originalTranscript: "",
          translatedTranscript: "",
          metadata: { durationMs: Date.now() - start, fromCache: false },
        };
      }

      // 2) Transcription avec Whisper
      // transcribeAudioFile = à implémenter: prend le fichier (path / buffer / URL) et renvoie { text, language }
      const transcription = await transcribeAudioFile(file);

      if (!transcription || !transcription.text || transcription.text.trim().length < 5) {
        return {
          success: false,
          error: "Transcription vide ou trop courte",
          sourceLang: transcription?.language || req.sourceLang || "unknown",
          targetLang: req.targetLang,
          originalTranscript: transcription?.text || "",
          translatedTranscript: "",
          metadata: { durationMs: Date.now() - start, fromCache: false },
        };
      }

      const sourceLang =
        req.sourceLang && req.sourceLang !== "auto"
          ? req.sourceLang
          : transcription.language || "auto";

      // 3) Traduction via Ulysse Translation Core
      const translation = await translationService.translate({
        text: transcription.text,
        targetLang: req.targetLang,
        sourceLang,
        domain: req.domain || "general",
        tone: req.tone || "neutral",
      });

      let audioFileId: number | undefined = undefined;

      // 4) Synthèse audio optionnelle
      if (req.generateAudio !== false) {
        const ttsResult = await synthesizeSpeech({
          userId: req.userId,
          text: translation.translated,
          lang: req.targetLang,
          voice: "default", // à gérer dans ton TTS (masc/fem, persona, etc.)
          sourceFileId: file.id,
        });

        if (ttsResult?.fileId) {
          audioFileId = ttsResult.fileId;
        }
      }

      return {
        success: true,
        sourceLang: translation.sourceLang || sourceLang,
        targetLang: translation.targetLang,
        originalTranscript: transcription.text,
        translatedTranscript: translation.translated,
        audioFileId,
        metadata: {
          durationMs: Date.now() - start,
          fromCache: translation.fromCache,
        },
      };
    } catch (error: any) {
      console.error("[AudioTranslate] Error:", error);
      return {
        success: false,
        error: error.message || "Erreur inconnue lors de la traduction audio",
        sourceLang: req.sourceLang || "unknown",
        targetLang: req.targetLang,
        originalTranscript: "",
        translatedTranscript: "",
        metadata: { durationMs: Date.now() - start, fromCache: false },
      };
    }
  }
}

export const audioTranslateService = new AudioTranslateService();
3️⃣ Les deux briques manquantes : transcribeAudioFile & synthesizeSpeech
a) Transcription Whisper (whisperService.ts)
Si tu as déjà un service pour la transcription (tu sembles l’utiliser pour les vocaux), il suffit de l’envelopper.

Esquisse :

// services/whisperService.ts
import OpenAI from "openai";
import { getFileStreamOrBuffer } from "./files"; // à adapter

const openai = new OpenAI({
  apiKey: process.env.AI_INTEGRATIONS_OPENAI_API_KEY,
  baseURL: process.env.AI_INTEGRATIONS_OPENAI_BASE_URL,
});

interface TranscriptionResult {
  text: string;
  language?: string;
}

export async function transcribeAudioFile(file: { id: number; storagePath?: string; url?: string }): Promise<TranscriptionResult> {
  // Récupérer le stream/buffer du fichier
  const audio = await getFileStreamOrBuffer(file);

  const response = await openai.audio.transcriptions.create({
    file: audio,         // selon SDK, peut être stream, buffer, etc.
    model: "whisper-1",
    response_format: "json",
  });

  return {
    text: response.text,
    language: response.language || undefined,
  };
}
Tu adapteras à ton système de fichiers (GCS / local / S3).
