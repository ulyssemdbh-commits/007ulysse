Voilà un voiceAuthService.ts complet, propre, prêt à intégrer dans ton Core.
Il encapsule tout: vérif speaker, niveaux d’auth, cache de session, policies par action.

// services/voiceAuthService.ts

import type { Buffer } from "buffer";
import {
  verifySpeaker,
  isSpeakerServiceAvailable,
  hasVoiceProfile,
} from "./speakerVerificationService"; // adapte le chemin si besoin

// Niveaux d'auth vocale
export type VoiceAuthLevel = "reject" | "limited" | "full";

// Actions sensibles pour lesquelles on décidera si la voix suffit ou pas
export type VoiceAction =
  | "generic_chat"
  | "private_info"
  | "sports_pronos"
  | "sugu_management"
  | "domotics_control"
  | "settings_change";

// Politique de niveau minimum par action
const VOICE_POLICY: Record<VoiceAction, VoiceAuthLevel> = {
  generic_chat: "limited",     // on peut répondre même si confiance moyenne
  private_info: "full",
  sports_pronos: "full",
  sugu_management: "full",
  domotics_control: "full",
  settings_change: "full",
};

// Cache de session vocale
interface VoiceSession {
  level: VoiceAuthLevel;
  confidence: number;
  lastVerifiedAt: number;
}

const voiceSessionCache = new Map<string /* userId */, VoiceSession>();
const VOICE_SESSION_TTL_MS = 30 * 1000; // 30s: plusieurs requêtes dans la foulée = pas de re-vérif

// Pour limiter les logs/effets si service down
let speakerServiceLastCheck = 0;
let speakerServiceAvailableCache = false;
const SPEAKER_SERVICE_CHECK_TTL_MS = 10 * 1000; // 10s

export interface VoiceAuthResult {
  allowed: boolean;         // l'action est-elle autorisée ?
  level: VoiceAuthLevel;    // niveau d'auth atteignable
  confidence: number;       // confiance speaker service
  reason?: string;          // explication courte
  serviceAvailable: boolean;
  profileReady: boolean;    // profil voix suffisamment entraîné ?
}

/**
 * Convertit un score de confiance brut en niveau d'auth.
 * Seuils pensés pour être PRO / exigeants.
 */
export function evaluateVoiceAuth(confidence: number): VoiceAuthLevel {
  if (confidence >= 0.85) return "full";
  if (confidence >= 0.65) return "limited";
  return "reject";
}

/**
 * Compare le niveau courant à ce que demande la politique pour l'action.
 */
export function isVoiceAllowedFor(action: VoiceAction, level: VoiceAuthLevel): boolean {
  const required = VOICE_POLICY[action];
  if (required === "full") return level === "full";
  if (required === "limited") return level === "full" || level === "limited";
  return true;
}

/**
 * Vérifie si le speaker service est dispo (avec petit cache).
 */
async function checkSpeakerService(): Promise<boolean> {
  const now = Date.now();
  if (now - speakerServiceLastCheck < SPEAKER_SERVICE_CHECK_TTL_MS) {
    return speakerServiceAvailableCache;
  }
  const available = await isSpeakerServiceAvailable();
  speakerServiceAvailableCache = available;
  speakerServiceLastCheck = now;
  return available;
}

/**
 * Vérifie si on a une session récente pour éviter de re-vérifier pour chaque requête audio.
 */
function getCachedSession(userId: string): VoiceSession | null {
  const sess = voiceSessionCache.get(userId);
  if (!sess) return null;
  if (Date.now() - sess.lastVerifiedAt > VOICE_SESSION_TTL_MS) {
    voiceSessionCache.delete(userId);
    return null;
  }
  return sess;
}

function setCachedSession(userId: string, level: VoiceAuthLevel, confidence: number) {
  voiceSessionCache.set(userId, {
    level,
    confidence,
    lastVerifiedAt: Date.now(),
  });
}

/**
 * Service principal: autoriser ou non une action basée sur la voix.
 *
 * - userId: ID backend de l'utilisateur authentifié
 * - audioBuffer: audio brut de la requête vocale (WebM, etc.)
 * - action: type d'action qu'on veut autoriser (SUGU, pronos, etc.)
 */
export async function authorizeVoiceAction(
  userId: string,
  audioBuffer: Buffer,
  action: VoiceAction
): Promise<VoiceAuthResult> {
  // 1) Check speaker service dispo
  const available = await checkSpeakerService();
  if (!available) {
    // Service down: à toi de décider la policy globale.
    // Ici: on n'autorise PAS les actions sensibles, mais on tolère le chat générique.
    if (action === "generic_chat") {
      return {
        allowed: true,
        level: "limited",
        confidence: 0,
        reason: "Speaker service indisponible, passage en mode vocal limité.",
        serviceAvailable: false,
        profileReady: false,
      };
    }
    return {
      allowed: false,
      level: "reject",
      confidence: 0,
      reason: "Service de reconnaissance vocale indisponible.",
      serviceAvailable: false,
      profileReady: false,
    };
  }

  // 2) Vérifier que le profil vocal est suffisamment entraîné
  const profileReady = await hasVoiceProfile(userId);
  if (!profileReady) {
    // Pas de profil → pas d'auth forte. On peut tolérer le chat générique mais rien de sensible.
    if (action === "generic_chat") {
      return {
        allowed: true,
        level: "limited",
        confidence: 0,
        reason: "Aucun profil vocal suffisamment entraîné. Chat autorisé, actions sensibles bloquées.",
        serviceAvailable: true,
        profileReady: false,
      };
    }
    return {
      allowed: false,
      level: "reject",
      confidence: 0,
      reason: "Profil vocal incomplet. Enrôle ta voix avant d'utiliser cette action.",
      serviceAvailable: true,
      profileReady: false,
    };
  }

  // 3) Vérifier si on a déjà une session valide très récente
  const cached = getCachedSession(userId);
  if (cached) {
    const allowed = isVoiceAllowedFor(action, cached.level);
    return {
      allowed,
      level: cached.level,
      confidence: cached.confidence,
      reason: allowed
        ? "Session vocale récente réutilisée."
        : "Niveau d'auth vocal insuffisant pour cette action (cache).",
      serviceAvailable: true,
      profileReady: true,
    };
  }

  // 4) Vérification vocale live
  const verification = await verifySpeaker(audioBuffer, userId);
  const level = evaluateVoiceAuth(verification.confidence);

  // Stocker en session pour les prochains 30s
  setCachedSession(userId, level, verification.confidence);

  const allowed = isVoiceAllowedFor(action, level);

  let reason: string;
  if (!verification.verified) {
    reason =
      level === "reject"
        ? verification.error || "Voix non reconnue comme celle de l'utilisateur."
        : "Voix partiellement reconnue, mais pas assez fiable.";
  } else {
    reason = `Voix reconnue avec confiance ${Math.round(
      verification.confidence * 100
    )}%.`;
  }

  if (!allowed) {
    return {
      allowed: false,
      level,
      confidence: verification.confidence,
      reason: `Niveau d'auth vocal insuffisant pour l'action "${action}". ${reason}`,
      serviceAvailable: true,
      profileReady: true,
    };
  }

  return {
    allowed: true,
    level,
    confidence: verification.confidence,
    reason,
    serviceAvailable: true,
    profileReady: true,
  };
}
