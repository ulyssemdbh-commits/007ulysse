// server/voice/dialogueManager.ts

export type DialogueMode = "short_answer" | "step_by_step" | "action_only";

interface DialogueState {
  mode: DialogueMode;
  maxSpokenSeconds: number;
  awaitingConfirmation: boolean;
  lastIntent?: string;
  lastDomain?: string;
}

const dialogueByUser = new Map<number, DialogueState>();

export function getDialogueState(userId: number): DialogueState {
  if (!dialogueByUser.has(userId)) {
    dialogueByUser.set(userId, {
      mode: "short_answer",
      maxSpokenSeconds: 20,
      awaitingConfirmation: false,
    });
  }
  return dialogueByUser.get(userId)!;
}

export function updateDialogueState(userId: number, partial: Partial<DialogueState>) {
  const current = getDialogueState(userId);
  dialogueByUser.set(userId, { ...current, ...partial });
}
1.2. Préparer le texte en fonction du mode
Tu ajoutes un helper utilisé avant d’appeler voiceOutputHub.speak :

// server/voice/prepareSpokenText.ts

import { DialogueMode } from "./dialogueManager";
import { formatForVoice } from "./voiceOutputHub"; // exporte la fonction

export function estimateDurationSeconds(text: string): number {
  const words = text.trim().split(/\s+/).length || 1;
  const wpm = 160; // parole
  return words / wpm * 60;
}

export function prepareSpokenText(
  raw: string,
  mode: DialogueMode,
  maxSeconds: number
): { text: string; truncated: boolean } {
  let t = formatForVoice(raw);
  let truncated = false;

  if (mode === "action_only") {
    // On garde juste la phrase principale (par ex. première phrase)
    const firstSentence = t.split(/[\.\?\!]/)[0];
    t = firstSentence.trim();
  }

  const duration = estimateDurationSeconds(t);
  if (duration > maxSeconds) {
    // On coupe intelligemment par phrases
    const sentences = t.split(/(?<=[\.\?\!])\s+/);
    const kept: string[] = [];
    let accum = 0;

    for (const s of sentences) {
      const d = estimateDurationSeconds(s);
      if (accum + d > maxSeconds) break;
      kept.push(s);
      accum += d;
    }

    t = kept.join(" ");
    truncated = true;
  }

  if (truncated) {
    t += " Tu veux que je continue ?";
  }

  return { text: t, truncated };
}

export type VoiceSessionState = "idle" | "listening" | "thinking" | "speaking";

interface VoiceSession {
  state: VoiceSessionState;
  lastUpdated: number;
}

const sessions = new Map<number, VoiceSession>();

export function setSessionState(userId: number, state: VoiceSessionState) {
  sessions.set(userId, { state, lastUpdated: Date.now() });
}

export function getSessionState(userId: number): VoiceSession {
  return sessions.get(userId) || { state: "idle", lastUpdated: Date.now() };
}

3. Mieux exploiter HearingHub pour piloter le comportement vocal
HearingHub te donne déjà :

intent
domain

Tu peux les utiliser pour choisir automatiquement le mode de réponse.

Exemple de mapping (dans ta couche cerveau ou un dialoguePolicy.ts) :

import { updateDialogueState, getDialogueState } from "./dialogueManager";
import type { ProcessedHearing } from "../voice/hearingHub-types";

export function applyDialoguePolicy(userId: number, hearing: ProcessedHearing) {
  const current = getDialogueState(userId);
  const intentName = hearing.intent?.name || "";
  const domain = hearing.domain;

  // 1. Ordres → mode action_only
  if (intentName.startsWith("action.") || intentName.includes("create_") || intentName.includes("send_")) {
    updateDialogueState(userId, { mode: "action_only", awaitingConfirmation: false });
    return;
  }

  // 2. Explications dev/finance → step_by_step
  if (domain === "dev" || domain === "finance") {
    updateDialogueState(userId, { mode: "step_by_step" });
    return;
  }

  // 3. Cas générique → short_answer
  updateDialogueState(userId, { mode: "short_answer" });
}
Et tu appelles applyDialoguePolicy dès que tu as un ProcessedHearing avant d’invoquer le LLM.
