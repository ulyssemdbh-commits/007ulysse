3.2. Service d’analyse de la requête
On généralise ta logique de codeContextOrchestrator.shouldInjectCodeContext:

// server/services/requestAnalysisService.ts

import { containsCodeKeywords, containsDiagnosticKeywords } from "./keywordUtils"; // à extraire
import { detectSportsKeywords } from "./sportsKeywordUtils"; // nouveau

export const requestAnalysisService = {
  analyze(userMessage: string, isOwner: boolean): RequestAnalysis {
    const lower = userMessage.toLowerCase();

    // Code
    if (isOwner && containsCodeKeywords(lower)) {
      const diag = containsDiagnosticKeywords(lower);
      return {
        domain: diag ? "dev_diagnostics" : "code",
        isOwner,
        requiresHighReliability: false
      };
    }

    // Sports / paris
    if (detectSportsKeywords(lower)) {
      return {
        domain: "sports",
        isOwner,
        requiresHighReliability: true // données chiffrées / prono
      };
    }

    // Par défaut
    return {
      domain: "general",
      isOwner,
      requiresHighReliability: false
    };
  }
};
(On pourra brancher ici plus tard un modèle de classification au lieu de mots-clés.)

3.3. UnifiedContextService – cœur du système
// server/services/unifiedContextService.ts

import { contextCacheService } from "./contextCacheService";
import { getPreloadedContext, startPreloading } from "./contextPreloader";
import { codeContextOrchestrator } from "./codeContextOrchestrator";
import { requestAnalysisService } from "./requestAnalysisService";
import { marsAuditContextService } from "./marsAuditContextService"; // nouveau petit service
import { db } from "../db";
import { users } from "@shared/schema";
import { eq } from "drizzle-orm";

export const unifiedContextService = {
  async getUnifiedContext(
    userId: number,
    userMessage: string
  ): Promise<UnifiedContext> {
    const isOwner = await this.isOwner(userId);
    const analysis = requestAnalysisService.analyze(userMessage, isOwner);

    // 1) Core context (cache 5 min)
    const coreContext = await contextCacheService.buildContext(userId, false);

    // 2) Live context – essayer cache preloader, sinon lancer en background
    const liveContext = await this.getLiveContext(userId);

    // 3) Domain context ciblé
    const domainContext = await this.getDomainContexts(userId, userMessage, analysis);

    const unified: UnifiedContext = {
      userId,
      analysis,
      core: {
        systemPrompt: coreContext.systemPrompt,
        recentMemories: coreContext.recentMemories,
        importantKnowledge: coreContext.importantKnowledge,
        devContext: coreContext.devContext
      },
      live: liveContext,
      domain: domainContext,
      builtAt: Date.now()
    };

    // 4) Notifier MARS audit (contexte utilisé)
    marsAuditContextService.recordContextSnapshot(unified);

    return unified;
  },

  async isOwner(userId: number): Promise<boolean> {
    const [user] = await db.select().from(users).where(eq(users.id, userId));
    return !!user?.isOwner;
  },

  async getLiveContext(userId: number): Promise<LiveContext> {
    const preloaded = getPreloadedContext(userId);

    // Si pas de cache → lancer un preload en background pour la prochaine requête
    if (!preloaded) {
      startPreloading(userId).catch(err =>
        console.error("[UnifiedContext] Preload failed:", err)
      );
      return {};
    }

    return {
      timeContext: preloaded.timeContext || undefined,
      calendarContext: preloaded.calendarContext || undefined,
      memoryContextFromPreloader: preloaded.memoryContext || undefined,
      spotifyContext: preloaded.spotifyContext || undefined,
      geolocationContext: preloaded.geolocationContext || undefined
    };
  },

  async getDomainContexts(
    userId: number,
    userMessage: string,
    analysis: RequestAnalysis
  ): Promise<DomainContexts> {
    const domain: DomainContexts = {};

    if (analysis.domain === "code" || analysis.domain === "dev_diagnostics") {
      const { context: codeCtx } = await codeContextOrchestrator.checkAndGetContextForMessage(
        userId,
        userMessage
      );
      if (codeCtx) domain.codeContext = codeCtx;
    }

    if (analysis.domain === "sports") {
      const sportsCtx = await this.getSportsContext(userId, userMessage);
      if (sportsCtx) domain.sportsContext = sportsCtx;
    }

    return domain;
  },

  async getSportsContext(userId: number, userMessage: string): Promise<string | null> {
    // Nouveau: service dédié, qui va taper dans ton cache Djedou Pronos, standings, etc.
    const { sportsContextBuilder } = await import("./sportsContextBuilder");
    return sportsContextBuilder.buildContextForMessage(userId, userMessage);
  }
};
3.4. MarsAuditContextService – pont avec les logs MARS
// server/services/marsAuditContextService.ts

import { metricsService } from "./metricsService";

export const marsAuditContextService = {
  recordContextSnapshot(ctx: UnifiedContext) {
    try {
      metricsService.recordContextSnapshot({
        userId: ctx.userId,
        domain: ctx.analysis.domain,
        hasCore: true,
        hasLiveTime: !!ctx.live.timeContext,
        hasLiveCalendar: !!ctx.live.calendarContext,
        hasLiveGeo: !!ctx.live.geolocationContext,
        hasCodeContext: !!ctx.domain.codeContext,
        hasSportsContext: !!ctx.domain.sportsContext,
        requiresHighReliability: ctx.analysis.requiresHighReliability
      });
    } catch (err) {
      console.error("[MARS_AUDIT] Failed to record context snapshot:", err);
    }
  }
};
