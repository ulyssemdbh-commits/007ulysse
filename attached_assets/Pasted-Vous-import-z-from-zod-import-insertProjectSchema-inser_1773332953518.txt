import { z } from 'zod';
import { api } from './api-spec';

type ApiSpec = typeof api;

type Endpoint<
  Resource extends keyof ApiSpec,
  Action extends keyof ApiSpec[Resource]
> = ApiSpec[Resource][Action];

type InferInput<E> = E extends { input: infer I } ? I : undefined;
type InferResponses<E> = E extends { responses: infer R } ? R : never;

type InferSuccessResponse<E> = InferResponses<E>[keyof InferResponses<E>];

async function callApi<
  Resource extends keyof ApiSpec,
  Action extends keyof ApiSpec[Resource]
>(
  resource: Resource,
  action: Action,
  options: {
    params?: Record<string, string | number>;
    body?: InferInput<Endpoint<Resource, Action>> extends z.ZodTypeAny
      ? z.infer<InferInput<Endpoint<Resource, Action>>>
      : never;
  },
): Promise<z.infer<InferSuccessResponse<Endpoint<Resource, Action>>>> {
  const endpoint = api[resource][action] as Endpoint<Resource, Action>;
  let url = endpoint.path;

  if (options?.params) {
    Object.entries(options.params).forEach(([key, value]) => {
      if (url.includes(`:${key}`)) {
        url = url.replace(`:${key}`, String(value));
      }
    });
  }

  const res = await fetch(url, {
    method: endpoint.method,
    headers: { 'Content-Type': 'application/json' },
    body: endpoint.method !== 'GET' && options.body
      ? JSON.stringify(options.body)
      : undefined,
  });

  const responses = endpoint.responses;
  const schema = responses[res.status as keyof typeof responses] as
    | z.ZodTypeAny
    | undefined;

  if (!schema) {
    throw new Error(`Unexpected status ${res.status} for ${String(resource)}.${String(action)}`);
  }

  const json = res.status === 204 ? undefined : await res.json();
  return schema.parse(json);
}
