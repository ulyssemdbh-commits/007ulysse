import { fetchDynamicPage } from "./dynamicFetch"; // ton module

async function getPronosoftLdc() {
  const url = "https://www.pronosoft.com/fr/parions_sport/imprimer-liste.php"; 
  // idéalement avec les filtres LDC dans la query

  const result = await fetchDynamicPage({
    url,
    mode: "dom",        // pour exécuter l'extractorScript pronosoft
    project: "ldc-parionssport"
  });

  if (!result.success || !result.data) {
    throw new Error("Pronosoft fetch failed");
  }

  const matches = result.data as {
    teams: string;
    odds: string[];
  }[];

  // Normalisation cotes
  const normalized = matches.map(m => {
    const [o1, oN, o2] = m.odds
      .map(o => parseFloat(o.replace(",", ".").trim()))
      .filter(n => !isNaN(n) && n > 1 && n < 100);

    return {
      teams: m.teams,   // ex: "Manchester City - Galatasaray"
      odd1: o1,
      oddN: oN,
      odd2: o2,
    };
  });

  return normalized;
}
