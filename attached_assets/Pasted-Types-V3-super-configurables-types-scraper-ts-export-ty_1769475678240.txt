export type ScrapeMode = 'auto' | 'http' | 'browser';
export type LinkScope = 'same-domain' | 'same-host' | 'same-path' | 'all' | 'none';

export interface SelectorConfig {
  selector: string;
  attr?: string;   // "href", "src", "data-*"... sinon texte
  multi?: boolean; // true = tableau, false = un seul
}

export interface ScrapeRequest {
  url: string;

  mode?: ScrapeMode;              // auto | http | browser
  depth?: number;                 // 0 = pas de crawl, >0 = profondeur max

  maxConcurrency?: number;        // parallélisme

  respectRobotsTxt?: boolean;     // hook futur si tu veux

const BLOCK_LOCALHOST_BY_DEFAULT = true;
const BLOCKED_HOSTS = ['localhost', '127.0.0.1', '0.0.0.0', '::1'];

function basicHostCheck(url: string): void {
  if (!BLOCK_LOCALHOST_BY_DEFAULT) return;
  const host = new URL(url).hostname.toLowerCase();
  if (BLOCKED_HOSTS.includes(host)) {
    throw new Error(`Host "${host}" bloqué par sécurité (configurable).`);
  }
}

function applyProfileDefaults(req: ScrapeRequest): ScrapeRequest {
  const clone: ScrapeRequest = {
    ...req,
    extract: { ...(req.extract || {}) },
  };

  switch (req.profile) {
    case 'news':
      clone.depth ??= 1;
      clone.maxPages ??= 40;
      clone.maxConcurrency ??= 4;
      clone.linkScope ??= 'same-host';
      clone.extract!.text ??= true;
      clone.extract!.links ??= true;
      clone.extract!.metadata ??= true;
      break;

    case 'foot':
      // idéal pour sites de classements / matchs
      clone.depth ??= 1;
      clone.maxPages ??= 60;
      clone.maxConcurrency ??= 4;
      clone.linkScope ??= 'same-host';
      clone.extract!.text ??= true;
      clone.extract!.links ??= true;
      clone.extract!.metadata ??= true;
      break;

    case 'ecommerce':
      clone.depth ??= 1;
      clone.maxPages ??= 80;
      clone.maxConcurrency ??= 5;
      clone.linkScope ??= 'same-host';
      clone.excludeExtensions ??= ['.jpg', '.jpeg', '.png', '.gif', '.webp'];
      clone.extract!.text ??= true;
      clone.extract!.links ??= true;
      clone.extract!.metadata ??= true;
      break;

    case 'default':
    default:
      // tu peux ajouter d’autres profils custom si tu veux
      break;
  }

  return clone;
}

function normalizeUrl(base: string, href: string): string | null {
  try {
    if (!href || href.startsWith('#') || href.startsWith('javascript:')) return null;
    return new URL(href, base).toString();
  } catch {
    return null;
  }
}

function shouldFollowLink(
  currentUrl: string,
  targetUrl: string,
  scope: LinkScope | undefined,
): boolean {
  if (!scope || scope === 'none') return false;
  const from = new URL(currentUrl);
  const to = new URL(targetUrl);

  switch (scope) {
    case 'same-domain':
      return from.hostname.split('.').slice(-2).join('.') ===
        to.hostname.split('.').slice(-2).join('.');
    case 'same-host':
      return from.hostname === to.hostname;
    case 'same-path':
      return from.hostname === to.hostname &&
        to.pathname.startsWith(from.pathname);
    case 'all':
      return true;
    default:
      return false;
  }
}

function isExtensionExcluded(url: string, exts?: string[]): boolean {
  if (!exts || exts.length === 0) return false;
  const pathname = new URL(url).pathname.toLowerCase();
  return exts.some((ext) => pathname.endsWith(ext.toLowerCase()));
}

function parseSelectorConfig(raw: string | SelectorConfig): SelectorConfig {
  if (typeof raw === 'string') return { selector: raw };
  return raw;
}

  const page: ScrapedPage = {
    url,
    status: 200,
  };

  if (extract?.metadata) {
    page.metadata = {
      title: $('title').first().text().trim() || undefined,
      contentType,
    };
  }

  if (extract?.text) {
    page.text = $('body')
      .text()
      .replace(/\s+/g, ' ')
      .trim();
  }

  if (extract?.html) {
    page.html = body;
  }

  if (extract?.links) {
    const links: string[] = [];
    $('a[href]').each((_, el) => {
      const href = $(el).attr('href');
      const absolute = href ? normalizeUrl(url, href) : null;
      if (absolute) links.push(absolute);
    });
    page.links = [...new Set(links)];
