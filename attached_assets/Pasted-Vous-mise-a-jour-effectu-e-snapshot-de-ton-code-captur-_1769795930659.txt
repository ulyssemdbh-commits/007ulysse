private detectConnection(source: any, target: any): { type: string; strength: number; context: string } | null {
  const sourceTags = (source.tags || []) as string[];
  const targetTags = (target.tags || []) as string[];
  const sourceContent = `${source.title} ${source.content}`.toLowerCase();
  const targetContent = `${target.title} ${target.content}`.toLowerCase();

  const importanceA = typeof source.importance === "number" ? source.importance : 50;
  const importanceB = typeof target.importance === "number" ? target.importance : 50;
  const importanceFactor = (importanceA + importanceB) / 2; // 0-100

  const weightBase = 0.6;
  const weightImportance = 0.4;

  const combine = (base: number) =>
    Math.min(
      100,
      Math.round(base * weightBase + importanceFactor * weightImportance)
    );

  const commonTags = sourceTags.filter(t => targetTags.includes(t));
  if (commonTags.length > 0) {
    const baseStrength = Math.min(100, commonTags.length * 30);
    return {
      type: "related_by_tag",
      strength: combine(baseStrength),
      context: `Tags communs: ${commonTags.join(", ")}`,
    };
  }

  if (source.category === target.category) {
    const subcategoryMatch = source.subcategory === target.subcategory;
    const baseStrength = subcategoryMatch ? 70 : 40;
    return {
      type: subcategoryMatch ? "same_topic" : "same_category",
      strength: combine(baseStrength),
      context: `Même catégorie: ${source.category}${subcategoryMatch ? ` / ${source.subcategory}` : ""}`,
    };
  }

  const sourceWords = new Set(
    sourceContent.split(/\s+/).filter((w: string) => w.length > 4)
  );
  const targetWords = new Set(
    targetContent.split(/\s+/).filter((w: string) => w.length > 4)
  );
  let commonWords = 0;
  sourceWords.forEach(word => {
    if (targetWords.has(word)) commonWords++;
  });

  if (commonWords >= 3) {
    const baseStrength = Math.min(80, commonWords * 10);
    return {
      type: "semantic_similarity",
      strength: combine(baseStrength),
      context: `${commonWords} mots-clés communs`,
    };
  }

  return null;
}
