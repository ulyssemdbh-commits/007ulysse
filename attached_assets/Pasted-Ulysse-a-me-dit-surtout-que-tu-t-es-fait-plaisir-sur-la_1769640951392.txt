// --- routingProvider.ts (ou bloc en haut du fichier) ---

type RoutingProfile = "driving" | "walking" | "cycling";

interface OSRMRoute {
  distance: number; // meters
  duration: number; // seconds
  geometry?: string;
}

type CacheValue =
  | { kind: "distance"; distance: number; duration: number }
  | { kind: "matrix"; distances: number[][] };

interface CacheEntry {
  value: CacheValue;
  timestamp: number;
}

const routingCache = new Map<string, CacheEntry>();
const CACHE_TTL_MS = 60 * 60 * 1000; // 1h
const MAX_CACHE_ENTRIES = 500;

const OSRM_BASE_URL = process.env.OSRM_BASE_URL || "https://router.project-osrm.org";

function makeRouteKey(fromLat: number, fromLng: number, toLat: number, toLng: number, profile: RoutingProfile) {
  return `route:${profile}:${fromLat.toFixed(5)},${fromLng.toFixed(5)}->${toLat.toFixed(5)},${toLng.toFixed(5)}`;
}

function makeMatrixKey(waypoints: { lat: number; lng: number }[], profile: RoutingProfile) {
  const coords = waypoints.map(wp => `${wp.lng.toFixed(5)},${wp.lat.toFixed(5)}`).join(";");
  return `matrix:${profile}:${coords}`;
}

function getCache(key: string, kind: CacheValue["kind"]): CacheValue | null {
  const entry = routingCache.get(key);
  if (!entry) return null;
  if (Date.now() - entry.timestamp > CACHE_TTL_MS) {
    routingCache.delete(key);
    return null;
  }
  if (entry.value.kind !== kind) return null;
  return entry.value;
}

function setCache(key: string, value: CacheValue) {
  if (routingCache.size >= MAX_CACHE_ENTRIES) {
    // stratégie simple: supprimer des entrées les plus anciennes
    const entries = Array.from(routingCache.entries()).sort((a, b) => a[1].timestamp - b[1].timestamp);
    for (let i = 0; i < Math.min(50, entries.length); i++) {
      routingCache.delete(entries[i][0]);
    }
  }
  routingCache.set(key, { value, timestamp: Date.now() });
}

async function fetchWithTimeout(url: string, timeoutMs: number): Promise<Response> {
  const controller = new AbortController();
  const timeoutId = setTimeout(() => controller.abort(), timeoutMs);
  try {
    const res = await fetch(url, { signal: controller.signal });
    return res;
  } finally {
    clearTimeout(timeoutId);
  }
}

export class RoutingProvider {
  constructor(private baseUrl: string = OSRM_BASE_URL) {}

  async getRoute(
    fromLat: number,
    fromLng: number,
    toLat: number,
    toLng: number,
    profile: RoutingProfile = "driving",
    retries = 3
  ): Promise<OSRMRoute | null> {
    const key = makeRouteKey(fromLat, fromLng, toLat, toLng, profile);
    const cached = getCache(key, "distance");
    if (cached) {
      const c = cached as { kind: "distance"; distance: number; duration: number };
      return { distance: c.distance, duration: c.duration };
    }

    for (let attempt = 0; attempt < retries; attempt++) {
      try {
        const url = `${this.baseUrl}/route/v1/${profile}/${fromLng},${fromLat};${toLng},${toLat}?overview=false`;
        const response = await fetchWithTimeout(url, 5000);

        if (!response.ok) throw new Error(`OSRM route returned ${response.status}`);
        const data = await response.json();

        if (data.code === "Ok" && data.routes && data.routes.length > 0) {
          const route = data.routes[0];
          setCache(key, { kind: "distance", distance: route.distance, duration: route.duration });
          return { distance: route.distance, duration: route.duration };
        }
        return null;
      } catch (err: any) {
        console.log(`[RoutingProvider] route attempt ${attempt + 1}/${retries} failed: ${err?.message}`);
        if (attempt < retries - 1) {
          await new Promise(r => setTimeout(r, 500 * (attempt + 1)));
        }
      }
    }
    return null;
  }

  async getMatrix(
    waypoints: { lat: number; lng: number }[],
    profile: RoutingProfile = "driving"
  ): Promise<number[][] | null> {
    const n = waypoints.length;
    if (n === 0) return [];
    const key = makeMatrixKey(waypoints, profile);
    const cached = getCache(key, "matrix");
    if (cached) {
      return (cached as { kind: "matrix"; distances: number[][] }).distances;
    }

    for (let attempt = 0; attempt < 3; attempt++) {
      try {
        const coords = waypoints.map(wp => `${wp.lng},${wp.lat}`).join(";");
        const url = `${this.baseUrl}/table/v1/${profile}/${coords}?annotations=distance`;
        const response = await fetchWithTimeout(url, 10000);

        if (!response.ok) throw new Error(`OSRM table returned ${response.status}`);
        const data = await response.json();

        if (data.code === "Ok" && data.distances) {
          setCache(key, { kind: "matrix", distances: data.distances });
          console.log(`[RoutingProvider] Table API returned ${n}x${n} matrix`);
          return data.distances;
        }
      } catch (err: any) {
        console.log(`[RoutingProvider] table attempt ${attempt + 1}/3 failed: ${err?.message}`);
        if (attempt < 2) {
          await new Promise(r => setTimeout(r, 500 * (attempt + 1)));
        }
      }
    }

    return null;
  }
}

// --- itineraryPlanner.ts (ou bloc dans le même fichier) ---

interface WaypointInput {
  lat: number;
  lng: number;
  label: string;
  address?: string;
  name?: string;
  notes?: string;
  isCurrentLocation?: boolean;
}

interface OptimizedRoute {
  waypoints: WaypointInput[];
  totalDistance: number;
  totalDuration: number;
  savings?: { distance: number; duration: number };
}

class ItineraryPlanner {
  constructor(private routing: RoutingProvider) {}

  private haversineDistance(lat1: number, lng1: number, lat2: number, lng2: number): number {
    const R = 6371000;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLng = (lng2 - lng1) * Math.PI / 180;
    const a = Math.sin(dLat / 2) ** 2 +
      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
      Math.sin(dLng / 2) ** 2;
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
  }

  private nearestNeighborWithMatrix(distMatrix: number[][], n: number): number[] {
    if (n <= 2) return n === 2 ? [0, 1] : [0];
    const result: number[] = [0];
    const remaining = new Set(Array.from({ length: n - 2 }, (_, i) => i + 1));
    let current = 0;

    while (remaining.size > 0) {
      let nearest = -1;
      let nearestDist = Infinity;
      for (const idx of Array.from(remaining)) {
        if (distMatrix[current][idx] < nearestDist) {
          nearestDist = distMatrix[current][idx];
          nearest = idx;
        }
      }
      if (nearest !== -1) {
        result.push(nearest);
        remaining.delete(nearest);
        current = nearest;
      } else {
        break;
      }
    }

    result.push(n - 1);
    return result;
  }

  private twoOptSwap(order: number[], i: number, j: number): number[] {
    const newOrder = [...order];
    while (i < j) {
      [newOrder[i], newOrder[j]] = [newOrder[j], newOrder[i]];
      i++; j--;
    }
    return newOrder;
  }

  private calculateTourDistance(order: number[], distMatrix: number[][]): number {
    let total = 0;
    for (let i = 0; i < order.length - 1; i++) {
      total += distMatrix[order[i]][order[i + 1]];
    }
    return total;
  }

  private async calculateRealDistance(
    waypoints: WaypointInput[],
    profile: RoutingProfile = "driving"
  ): Promise<number> {
    let total = 0;
    for (let i = 0; i < waypoints.length - 1; i++) {
      const route = await this.routing.getRoute(
        waypoints[i].lat, waypoints[i].lng,
        waypoints[i + 1].lat, waypoints[i + 1].lng,
        profile
      );
      total += route?.distance ?? this.haversineDistance(
        waypoints[i].lat, waypoints[i].lng,
        waypoints[i + 1].lat, waypoints[i + 1].lng
      );
    }
    return total;
  }

  async optimizeWaypointOrder(
    waypoints: WaypointInput[],
    profile: RoutingProfile = "driving"
  ): Promise<OptimizedRoute> {
    if (waypoints.length <= 2) {
      const totalDist = waypoints.length === 2
        ? await this.calculateRealDistance(waypoints, profile)
        : 0;
      const avgSpeed = profile === "walking" ? 1.4 : profile === "cycling" ? 4.5 : 11;
      return {
        waypoints,
        totalDistance: totalDist,
        totalDuration: Math.round(totalDist / avgSpeed),
      };
    }

    const start = waypoints[0];
    const end = waypoints[waypoints.length - 1];
    const middle = waypoints.slice(1, -1);

    console.log(`[Planner] Optimizing ${waypoints.length} points, profile=${profile}`);

    const allPoints = [start, ...middle, end];
    const matrixFromOSRM = await this.routing.getMatrix(allPoints, profile);

    let distMatrix: number[][];

    if (matrixFromOSRM) {
      distMatrix = matrixFromOSRM;
    } else {
      console.log("[Planner] Falling back to Haversine matrix");
      const n = allPoints.length;
      distMatrix = Array(n).fill(null).map(() => Array(n).fill(0));
      for (let i = 0; i < n; i++) {
        for (let j = 0; j < n; j++) {
          if (i !== j) {
            distMatrix[i][j] = this.haversineDistance(
              allPoints[i].lat, allPoints[i].lng,
              allPoints[j].lat, allPoints[j].lng
            );
          }
        }
      }
    }

    let bestOrder = this.nearestNeighborWithMatrix(distMatrix, allPoints.length);
    let bestDistance = this.calculateTourDistance(bestOrder, distMatrix);

    if (middle.length > 1) {
      let improved = true;
      let iterations = 0;
      const maxIterations = 100;

      while (improved && iterations < maxIterations) {
        improved = false;
        iterations++;
        for (let i = 1; i < bestOrder.length - 2; i++) {
          for (let j = i + 1; j < bestOrder.length - 1; j++) {
            const newOrder = this.twoOptSwap(bestOrder, i, j);
            const newDistance = this.calculateTourDistance(newOrder, distMatrix);
            if (newDistance < bestDistance - 0.1) {
              bestOrder = newOrder;
              bestDistance = newDistance;
              improved = true;
            }
          }
        }
      }
      console.log(`[Planner] 2-opt finished after ${iterations} iterations`);
    }

    const optimized = bestOrder.map((idx, i) => ({
      ...allPoints[idx],
      label: String.fromCharCode(65 + i),
    }));

    const originalDistance = await this.calculateRealDistance(waypoints, profile);
    const avgSpeed = profile === "walking" ? 1.4 : profile === "cycling" ? 4.5 : 11;

    return {
      waypoints: optimized,
      totalDistance: Math.round(bestDistance),
      totalDuration: Math.round(bestDistance / avgSpeed),
      savings: {
        distance: Math.round(originalDistance - bestDistance),
        duration: Math.round((originalDistance - bestDistance) / avgSpeed),
      },
    };
  }
}

Je te le mets en gardant ton modèle WaypointInput et en prenant nav + prefs en paramètre.

// --- navigationEngine.ts (ou bloc) ---

interface RoutePreferencesModel {
  deviationThreshold?: number;
  autoRecalculate?: boolean;
  arrivalAlertDistance?: number;
}

interface ActiveNavModel {
  profile: RoutingProfile;
  waypointsData: WaypointInput[];
  currentWaypointIndex: number;
  isOffRoute?: boolean;
  offRouteCount?: number;
  totalDistance?: number;
  totalDuration?: number;
  startedAt?: Date;
  // ... autres champs si besoin
}

class NavigationEngine {
  constructor(private updateNav: (data: Partial<ActiveNavigation>) => Promise<void>) {}

  private haversine(lat1: number, lng1: number, lat2: number, lng2: number): number {
    const R = 6371000;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLng = (lng2 - lng1) * Math.PI / 180;
    const a = Math.sin(dLat / 2) ** 2 +
      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
      Math.sin(dLng / 2) ** 2;
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
  }

  private pointToSegmentDistance(
    pointLat: number, pointLng: number,
    segStartLat: number, segStartLng: number,
    segEndLat: number, segEndLng: number
  ): number {
    const latScale = 111320;
    const lngScale = 111320 * Math.cos(pointLat * Math.PI / 180);

    const px = (pointLng - segStartLng) * lngScale;
    const py = (pointLat - segStartLat) * latScale;
    const sx = (segEndLng - segStartLng) * lngScale;
    const sy = (segEndLat - segStartLat) * latScale;

    const segLenSq = sx * sx + sy * sy;
    if (segLenSq === 0) return Math.sqrt(px * px + py * py);

    const t = Math.max(0, Math.min(1, (px * sx + py * sy) / segLenSq));
    const closestX = t * sx;
    const closestY = t * sy;

    return Math.sqrt((px - closestX) ** 2 + (py - closestY) ** 2);
  }

  async checkDeviation(
    userId: number,
    nav: ActiveNavModel,
    prefs: RoutePreferencesModel | null,
    currentLat: number,
    currentLng: number
  ) {
    const threshold = prefs?.deviationThreshold || 50;
    const waypoints = nav.waypointsData;
    const currentIndex = nav.currentWaypointIndex;

    if (currentIndex >= waypoints.length - 1) {
      return {
        isOffRoute: false,
        distanceFromRoute: 0,
        shouldRecalculate: false,
        currentSegment: waypoints.length - 1,
        progressPercent: 100,
        estimatedArrival: null,
      };
    }

    let minDistance = Infinity;
    let closestSegment = currentIndex;

    for (let i = currentIndex; i < waypoints.length - 1; i++) {
      const segDist = this.pointToSegmentDistance(
        currentLat, currentLng,
        waypoints[i].lat, waypoints[i].lng,
        waypoints[i + 1].lat, waypoints[i + 1].lng
      );
      if (segDist < minDistance) {
        minDistance = segDist;
        closestSegment = i;
      }
    }

    const distToNext = this.haversine(
      currentLat, currentLng,
      waypoints[currentIndex + 1].lat, waypoints[currentIndex + 1].lng
    );

    const isOffRoute = minDistance > threshold;
    const shouldRecalculate = isOffRoute && (prefs?.autoRecalculate !== false);

    let completedDistance = 0;
    let totalDistance = 0;
    for (let i = 0; i < waypoints.length - 1; i++) {
      const segDist = this.haversine(
        waypoints[i].lat, waypoints[i].lng,
        waypoints[i + 1].lat, waypoints[i + 1].lng
      );
      totalDistance += segDist;
      if (i < closestSegment) {
        completedDistance += segDist;
      } else if (i === closestSegment) {
        const segProgress = 1 - (distToNext / segDist);
        completedDistance += segDist * Math.max(0, Math.min(1, segProgress));
      }
    }
    const progressPercent = totalDistance > 0 ? Math.round((completedDistance / totalDistance) * 100) : 0;

    const remainingDistance = totalDistance - completedDistance;
    const avgSpeedMps = nav.profile === "walking" ? 1.4 : nav.profile === "cycling" ? 4.5 : 11;
    const remainingSeconds = remainingDistance / avgSpeedMps;
    const eta = new Date(Date.now() + remainingSeconds * 1000);
    const estimatedArrival = eta.toLocaleTimeString("fr-FR", { hour: "2-digit", minute: "2-digit" });

    if (isOffRoute) {
      await this.updateNav({
        isOffRoute: true,
        offRouteCount: (nav.offRouteCount || 0) + 1,
        lastKnownLat: currentLat.toString(),
        lastKnownLng: currentLng.toString(),
      });
      console.log(`[Navigation] User ${userId} off route by ${Math.round(minDistance)}m (threshold: ${threshold}m)`);
    } else if (nav.isOffRoute) {
      await this.updateNav({ isOffRoute: false });
      console.log(`[Navigation] User ${userId} back on route`);
    }

    return {
      isOffRoute,
      distanceFromRoute: Math.round(minDistance),
      shouldRecalculate,
      currentSegment: closestSegment,
      progressPercent,
      estimatedArrival,
    };
  }

  async checkArrivalProximity(
    nav: ActiveNavModel,
    prefs: RoutePreferencesModel | null,
    currentLat: number,
    currentLng: number
  ) {
    const waypoints = nav.waypointsData;
    const currentIndex = nav.currentWaypointIndex;
    const alertDistance = prefs?.arrivalAlertDistance || 200;

    if (currentIndex >= waypoints.length - 1) {
      const lastWp = waypoints[waypoints.length - 1];
      const dist = this.haversine(currentLat, currentLng, lastWp.lat, lastWp.lng);
      return {
        nearWaypoint: lastWp,
        waypointIndex: waypoints.length - 1,
        distanceToWaypoint: dist,
        shouldAlert: dist <= alertDistance,
      };
    }

    const nextWp = waypoints[currentIndex + 1];
    const dist = this.haversine(currentLat, currentLng, nextWp.lat, nextWp.lng);

    if (dist <= 30) {
      await this.updateNav({ currentWaypointIndex: currentIndex + 1 });
    }

    return {
      nearWaypoint: nextWp,
      waypointIndex: currentIndex + 1,
      distanceToWaypoint: dist,
      shouldAlert: dist <= alertDistance,
    };
  }

  calculateBearing(lat1: number, lng1: number, lat2: number, lng2: number): number {
    const toRad = (d: number) => d * Math.PI / 180;
    const toDeg = (r: number) => r * 180 / Math.PI;
    const dLng = toRad(lng2 - lng1);
    const lat1Rad = toRad(lat1);
    const lat2Rad = toRad(lat2);

    const y = Math.sin(dLng) * Math.cos(lat2Rad);
    const x = Math.cos(lat1Rad) * Math.sin(lat2Rad) -
      Math.sin(lat1Rad) * Math.cos(lat2Rad) * Math.cos(dLng);

    return (toDeg(Math.atan2(y, x)) + 360) % 360;
  }
}

// --- overpassClient.ts (ou bloc) ---

interface SpeedLimitInfo {
  speedLimit: number | null;
  roadName: string | null;
  roadType: string | null;
}

interface SpeedCameraInfo {
  lat: number;
  lng: number;
  type: string;
  distance: number;
}

interface SchoolZoneInfo {
  lat: number;
  lng: number;
  name: string;
  distance: number;
}

interface POIInfo {
  lat: number;
  lng: number;
  name: string;
  type: string;
  distance: number;
}

class OverpassClient {
  private async fetchOverpass(query: string, timeoutMs: number = 5000): Promise<any | null> {
    try {
      const controller = new AbortController();
      const timeoutId = setTimeout(() => controller.abort(), timeoutMs);
      const response = await fetch("https://overpass-api.de/api/interpreter", {
        method: "POST",
        body: `data=${encodeURIComponent(query)}`,
        headers: { "Content-Type": "application/x-www-form-urlencoded" },
        signal: controller.signal,
      });
      clearTimeout(timeoutId);
      if (!response.ok) return null;
      return await response.json();
    } catch (err) {
      console.log("[Overpass] request failed:", err);
      return null;
    }
  }

  private haversine(lat1: number, lng1: number, lat2: number, lng2: number): number {
    const R = 6371000;
    const dLat = (lat2 - lat1) * Math.PI / 180;
    const dLng = (lng2 - lng1) * Math.PI / 180;
    const a = Math.sin(dLat / 2) ** 2 +
      Math.cos(lat1 * Math.PI / 180) * Math.cos(lat2 * Math.PI / 180) *
      Math.sin(dLng / 2) ** 2;
    const c = 2 * Math.atan2(Math.sqrt(a), Math.sqrt(1 - a));
    return R * c;
  }

  async getSpeedLimit(lat: number, lng: number): Promise<SpeedLimitInfo> {
    const radius = 30;
    const query = `
      [out:json][timeout:5];
      way(around:${radius},${lat},${lng})["highway"];
      out body;
    `;
    const data = await this.fetchOverpass(query, 5000);
    if (!data || !data.elements || data.elements.length === 0) {
      return { speedLimit: null, roadName: null, roadType: null };
    }

    const road = data.elements[0];
    const tags = road.tags || {};
    let speedLimit: number | null = null;

    if (tags.maxspeed) {
      const match = tags.maxspeed.match(/^(\d+)/);
      if (match) {
        speedLimit = parseInt(match[1], 10);
        if (tags.maxspeed.includes("mph")) {
          speedLimit = Math.round(speedLimit * 1.60934);
        }
      }
    } else {
      const defaultSpeeds: Record<string, number> = {
        motorway: 130,
        trunk: 110,
        primary: 80,
        secondary: 70,
        tertiary: 50,
        residential: 30,
        living_street: 20,
        unclassified: 50,
      };
      speedLimit = defaultSpeeds[tags.highway] || null;
    }

    return {
      speedLimit,
      roadName: tags.name || tags.ref || null,
      roadType: tags.highway || null,
    };
  }

  async getSpeedCameras(lat: number, lng: number, radius: number = 1000): Promise<SpeedCameraInfo[]> {
    const query = `
      [out:json][timeout:5];
      (
        node["highway"="speed_camera"](around:${radius},${lat},${lng});
        node["enforcement"="maxspeed"](around:${radius},${lat},${lng});
        node["traffic_enforcement"](around:${radius},${lat},${lng});
      );
      out body;
    `;
    const data = await this.fetchOverpass(query, 5000);
    const res: SpeedCameraInfo[] = [];
    if (data && data.elements) {
      for (const el of data.elements) {
        const d = this.haversine(lat, lng, el.lat, el.lon);
        res.push({
          lat: el.lat,
          lng: el.lon,
          type: el.tags?.enforcement || el.tags?.highway || "speed_camera",
          distance: Math.round(d),
        });
      }
    }
    return res.sort((a, b) => a.distance - b.distance);
  }

  async getSchoolZones(lat: number, lng: number, radius: number = 500): Promise<SchoolZoneInfo[]> {
    const query = `
      [out:json][timeout:5];
      (
        way["amenity"="school"](around:${radius},${lat},${lng});
        node["amenity"="school"](around:${radius},${lat},${lng});
      );
      out center;
    `;
    const data = await this.fetchOverpass(query, 5000);
    const res: SchoolZoneInfo[] = [];
    if (data && data.elements) {
      for (const el of data.elements) {
        const elLat = el.center?.lat || el.lat;
        const elLng = el.center?.lon || el.lon;
        if (!elLat || !elLng) continue;
        const d = this.haversine(lat, lng, elLat, elLng);
        res.push({
          lat: elLat,
          lng: elLng,
          name: el.tags?.name || "École",
          distance: Math.round(d),
        });
      }
    }
    return res.sort((a, b) => a.distance - b.distance).slice(0, 5);
  }

  async getPOIs(lat: number, lng: number, radius: number = 2000): Promise<POIInfo[]> {
    const query = `
      [out:json][timeout:8];
      (
        node["amenity"="fuel"](around:${radius},${lat},${lng});
        node["amenity"="charging_station"](around:${radius},${lat},${lng});
        node["highway"="rest_area"](around:${radius},${lat},${lng});
        node["highway"="services"](around:${radius},${lat},${lng});
        node["amenity"="parking"](around:${radius},${lat},${lng});
      );
      out body;
    `;
    const data = await this.fetchOverpass(query, 8000);
    const res: POIInfo[] = [];
    if (data && data.elements) {
      const typeLabels: Record<string, string> = {
        fuel: "Station-service",
        charging_station: "Borne de recharge",
        rest_area: "Aire de repos",
        services: "Aire de service",
        parking: "Parking",
      };
      for (const el of data.elements) {
        const d = this.haversine(lat, lng, el.lat, el.lon);
        const type = el.tags?.amenity || el.tags?.highway || "poi";
        res.push({
          lat: el.lat,
          lng: el.lon,
          name: el.tags?.name || el.tags?.brand || typeLabels[type] || type,
          type,
          distance: Math.round(d),
        });
      }
    }
    return res.sort((a, b) => a.distance - b.distance).slice(0, 10);
  }
}
