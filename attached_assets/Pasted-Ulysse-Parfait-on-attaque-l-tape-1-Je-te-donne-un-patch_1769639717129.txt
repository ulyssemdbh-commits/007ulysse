const analyzeEchoSuspicion = useCallback((text: string): EchoAnalysis => {
  const reasons: string[] = [];
  let score = 0;

  // Si on ne parle pas, pas d'echo
  if (!isSpeakingRef.current) {
    return { score: 0, reasons };
  }

  const now = Date.now();
  const timeSinceTTSEnd = now - ttsEndTimeRef.current;

  // 1) Cooldown très proche de la fin TTS
  //    → suspicion moyenne, qu'on nuancera ensuite
  if (timeSinceTTSEnd < 300) {
    score += 0.35;
    reasons.push("cooldown");
  }

  // 2) Similarité de contenu avec le dernier texte parlé
  if (checkContentSimilarity(text)) {
    score += 0.35;
    reasons.push("content");
  }

  // 3) Analyse d'énergie si on a un analyser dispo
  if (analyserRef.current) {
    const energy = getAudioEnergy(analyserRef.current);

    // On laisse la logique fine à isEchoFeedback (qui connaît le seuil TTS)
    if (isEchoFeedback(energy)) {
      score += 0.3;
      reasons.push("energy");
    }

    // Baisse de suspicion si l'énergie est vraiment élevée (barge-in clair)
    if (energy > 30) {
      // L'utilisateur parle fort → on réduit la suspicion
      score -= 0.25;
      reasons.push("barge-in");
    }
  }

  // Clamp propre entre 0 et 1
  if (score < 0) score = 0;
  if (score > 1) score = 1;

  return { score, reasons };
}, [checkContentSimilarity, getAudioEnergy, isEchoFeedback]);
2) Remplacer shouldFilterTranscript
Remplace ton const shouldFilterTranscript = useCallback(... par cette version :

const shouldFilterTranscript = useCallback((text: string): boolean => {
  if (!isSpeakingRef.current) {
    echoFilterAppliedRef.current = false;
    return false;
  }

  const { score, reasons } = analyzeEchoSuspicion(text);
  const THRESHOLD = 0.75; // tu peux ajuster ce seuil

  if (score >= THRESHOLD) {
    logDebugEvent(
      `Echo filter: score=${score.toFixed(2)} reasons=${reasons.join("+") || "none"}`
    );
    echoFilterAppliedRef.current = true;
    return true;
  }

  echoFilterAppliedRef.current = false;
  return false;
}, [analyzeEchoSuspicion, logDebugEvent]);
