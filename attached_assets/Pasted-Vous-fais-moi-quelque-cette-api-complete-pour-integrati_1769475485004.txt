export type ScrapeMode = 'auto' | 'http' | 'browser';

export interface ScrapeRequest {
  url: string;
  mode?: ScrapeMode;           // "auto" (défaut) | "http" | "browser"
  depth?: number;              // 0 = pas de crawl, >0 = profondeur max
  maxPages?: number;           // sécurité anti-dérapage
  maxConcurrency?: number;     // nb de requêtes en parallèle
  userAgent?: string;          // override UA par défaut
  respectRobotsTxt?: boolean;  // par défaut true
  extract?: {
    text?: boolean;
    html?: boolean;
    links?: boolean;
    metadata?: boolean;
    selectors?: Record<string, string>; // nom -> sélecteur CSS
  };
}

  extracted?: Record<string, any>;
  error?: string;
}

export interface ScrapeResult {
  startedAt: string;
  finishedAt: string;

  pages: ScrapedPage[];

// services/scraperService.ts
import {
  CheerioCrawler,
  PlaywrightCrawler,
  log,
  RequestQueue,
} from 'crawlee';
import * as cheerio from 'cheerio';
import type { ScrapeRequest, ScrapeResult, ScrapedPage, ScrapeMode } from '../types/scraper';

log.setLevel(log.LEVELS.INFO);

const DEFAULT_USER_AGENT =
  'UlysseScraperBot/1.0 (+https://ulysseproject.org; respect robots.txt; contact: server-admin@local)';

function normalizeUrl(base: string, href: string): string | null {
  try {
    if (!href || href.startsWith('#') || href.startsWith('javascript:')) return null;
    return new URL(href, base).toString();
  } catch {
    return null;
  }
}

function extractWithCheerio(
  url: string,
  body: string,
  contentType: string | undefined,
  extract: ScrapeRequest['extract']
): ScrapedPage {
  const $ = cheerio.load(body);
  const page: ScrapedPage = {
    url,
    status: 200,
  };

  if (extract?.metadata) {
    page.metadata = {
      title: $('title').first().text().trim() || undefined,
      contentType,
    };
  }

  if (extract?.text) {
    page.text = $('body')
      .text()
      .replace(/\s+/g, ' ')
      .trim();
  }

  if (extract?.html) {
    page.html = body;
  }

  if (extract?.links) {
    const links: string[] = [];
    $('a[href]').each((_, el) => {
      const href = $(el).attr('href');
      const absolute = href ? normalizeUrl(url, href) : null;
      if (absolute) links.push(absolute);
    });
    page.links = [...new Set(links)];
  }

  if (extract?.selectors) {
    const result: Record<string, any> = {};
    for (const [key, selector] of Object.entries(extract.selectors)) {
      const elements = $(selector);
      if (elements.length === 0) {
        result[key] = null;
      } else if (elements.length === 1) {
        result[key] = elements.first().text().trim();
      } else {
        result[key] = elements
          .map((_, el) => $(el).text().trim())
          .get()
          .filter(Boolean);
      }
    }
    page.extracted = result;
  }

  return page;
}

export async function crawlAndScrape(req: ScrapeRequest): Promise<ScrapeResult> {
  const startedAt = new Date().toISOString();
  const pages: ScrapedPage[] = [];
  const visited = new Set<string>();

  const {
    url,
    mode = 'auto',
    depth = 0,
    maxPages = 50,
    maxConcurrency = 3,
    userAgent = DEFAULT_USER_AGENT,
    extract = { text: true, links: true, metadata: true },
    respectRobotsTxt = true,
  } = req;

  if (!url) {
    throw new Error('url is required');
  }

  const shouldFollowLinks = depth > 0;
  let modeUsed: ScrapeMode = mode;

  // ----- 1) HTTP mode (CheerioCrawler) -----
  const requestQueue = await RequestQueue.open();
  await requestQueue.addRequest({ url });

  const cheerioCrawler = new CheerioCrawler({
    requestQueue,
    maxRequestsPerCrawl: maxPages,
    maxConcurrency,
    requestHandlerTimeoutSecs: 60,
    useSessionPool: true,
    ignoreSslErrors: true,
    additionalMimeTypes: ['text/html', 'application/xhtml+xml'],
    // Respect basique des robots.txt via headers (le respect strict robots.txt est à ajouter si tu veux)
    preNavigationHooks: [
      async ({ request, session, proxyInfo }, gotoOptions) => {
        request.headers = {
          ...(request.headers || {}),
          'User-Agent': userAgent,
          Accept:
            'text/html,application/xhtml+xml,application/xml;q=0.9,*/*;q=0.8',
        };
        return gotoOptions;
      },
    ],
    async requestHandler({ request, response, body, contentType }) {
      if (pages.length >= maxPages) return;

      const currentUrl = request.loadedUrl || request.url;
      if (visited.has(currentUrl)) return;
      visited.add(currentUrl);

      if (!body || !contentType || !contentType.type.includes('text/html')) {
        pages.push({
          url: currentUrl,
          status: response?.statusCode ?? 0,
          error: 'Unsupported content type',
        });
        return;
      }

      const html = body.toString();
      const page = extractWithCheerio(
        currentUrl,
        html,
        contentType.type,
        extract
      );
      page.status = response?.statusCode ?? 200;

      pages.push(page);

      if (shouldFollowLinks && page.links) {
        const baseHost = new URL(url).hostname;

        for (const link of page.links) {
          if (pages.length >= maxPages) break;
          const linkHost = (() => {
            try {
              return new URL(link).hostname;
            } catch {
              return null;
            }
          })();
          if (!linkHost || linkHost !== baseHost) continue;
          if (!visited.has(link)) {
            await requestQueue.addRequest({ url: link });
          }
        }
      }
    },
    failedRequestHandler({ request, error }) {
      pages.push({
        url: request.url,
        status: 0,
        error: error?.message || 'Request failed',
      });
    },
  });

  // Si mode = http → on ne lance que ça
  if (mode === 'http') {
    await cheerioCrawler.run();
    modeUsed = 'http';
  } else if (mode === 'browser') {
    // ----- 2) Browser mode direct (Playwright) -----
    await runBrowserCrawler({
      url,
      maxPages,
      maxConcurrency,
      extract,
      userAgent,
      pages,
      visited,
      shouldFollowLinks,
    });
    modeUsed = 'browser';
  } else {
    // ----- 3) mode = auto → HTTP d'abord, fallback browser si on récupère quasi rien -----
    await cheerioCrawler.run();
    modeUsed = 'http';

    const hasUsefulContent =
      pages.length > 0 &&
      pages.some((p) => (p.text || '').length > 500 || (p.html || '').length > 2000);

    if (!hasUsefulContent) {
      log.info(
        `[Scraper] AUTO: HTTP content seems empty, falling back to browser mode for ${url}`
      );
      pages.length = 0;
      visited.clear();
      await runBrowserCrawler({
        url,
        maxPages,
        maxConcurrency,
        extract,
        userAgent,
        pages,
        visited,
        shouldFollowLinks,
      });
      modeUsed = 'browser';
    }
  }

  const finishedAt = new Date().toISOString();

  return {
    startedAt,
    finishedAt,
    modeUsed,
    pages,
  };
}
