On va commencer par un truc utile 90 % du temps :

À la fin de ta classe SportsCacheService, ajoute :

  /**
   * Recherche "intelligente" pour MARS sur les questions football
   * Exemple de queries:
   * - "grosse soirée LDC ce soir"
   * - "résultat OM hier"
   * - "matches Ligue 1 ce week-end"
   * - "top matchs du jour pour parier"
   */
  async searchQuery(rawQuery: string): Promise<{
    intent: "today_matches" | "week_matches" | "team_matches" | "competition_matches";
    teamName?: string;
    competition?: string;
    date?: Date;
    matches: any[];
  } | null> {
    const query = rawQuery.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "");

    // 1) Détecter la date (simplifié)
    const now = new Date();
    let targetDate: Date | undefined;

    if (query.includes("aujourd'hui") || query.includes("aujourdhui") || query.includes("ce soir") || query.includes("ce soir")) {
      targetDate = now;
    } else if (query.includes("demain")) {
      targetDate = new Date(now);
      targetDate.setDate(now.getDate() + 1);
    } else if (query.includes("hier")) {
      targetDate = new Date(now);
      targetDate.setDate(now.getDate() - 1);
    }

    // 2) Détecter compétition
    const competitionMap: Record<string, string> = {
      "ligue 1": "Ligue 1",
      "ligue1": "Ligue 1",
      "l1": "Ligue 1",
      "premier league": "Premier League",
      "pl": "Premier League",
      "la liga": "La Liga",
      "liga": "La Liga",
      "serie a": "Serie A",
      "bundesliga": "Bundesliga",
      "champions league": "Champions League",
      "ligue des champions": "Champions League",
      "ldc": "Champions League",
      "europa league": "Europa League",
    };

    let competition: string | undefined;
    for (const [key, value] of Object.entries(competitionMap)) {
      if (query.includes(key)) {
        competition = value;
        break;
      }
    }

    // 3) Détecter équipe (simple: on check dans les noms présents en DB aujourd'hui)
    let teamName: string | undefined;
    const knownTeamsToday = await this.getMatchesForDate(targetDate || now);
    const teamNames = new Set<string>();
    for (const m of knownTeamsToday) {
      if (m.homeTeam) teamNames.add(m.homeTeam);
      if (m.awayTeam) teamNames.add(m.awayTeam);
    }
    const normalizedTeams = Array.from(teamNames).map(name => ({
      name,
      norm: name.toLowerCase().normalize("NFD").replace(/[\u0300-\u036f]/g, "")
    }));

    const queryWords = query.split(/\s+/);

    for (const t of normalizedTeams) {
      if (query.includes(t.norm) || queryWords.some(w => t.norm.includes(w) && w.length >= 3)) {
        teamName = t.name;
        break;
      }
    }

    // 4) Déterminer l'intent
    // Cas 1 : matches du jour / soirée
    if (!teamName && !competition && targetDate) {
      const matches = await this.getMatchesWithOdds(targetDate);
      return {
        intent: "today_matches",
        date: targetDate,
        matches,
      };
    }

    // Cas 2 : compétition ciblée (sur semaine)
    if (competition) {
      const matches = await this.getUpcomingMatches(competition, 7);
      return {
        intent: "competition_matches",
        competition,
        matches,
      };
    }

    // Cas 3 : équipe ciblée
    if (teamName) {
      // pour l'instant, on prend les matches du jour avec cette équipe
      const matchesToday = await this.getMatchesWithOdds(targetDate || now);
      const teamMatches = matchesToday.filter(
        m =>
          m.homeTeam === teamName ||
          m.awayTeam === teamName
      );
      return {
        intent: "team_matches",
        teamName,
        date: targetDate,
        matches: teamMatches,
      };
    }

    // Cas 4 : matches de la semaine (fallback)
    const weekMatches = await this.getMatchesWithOdds(now);
    return {
      intent: "week_matches",
      matches: weekMatches,
    };
  }

Il reste deux helpers, mais ils sont côté “présentation” :

function buildSportsFactsFromData(sportsResult: {
  intent: string;
  matches: any[];
  teamName?: string;
  competition?: string;
  date?: Date;
}): FactAggregation["facts"] {
  const facts: FactAggregation["facts"] = [];

  for (const match of sportsResult.matches.slice(0, 10)) {
    const dateStr = new Date(match.matchDate).toLocaleString("fr-FR", {
      day: "2-digit",
      month: "2-digit",
      hour: "2-digit",
      minute: "2-digit",
    });

    // Si le match est terminé
    if (match.status === "finished" && match.homeScore != null && match.awayScore != null) {
      facts.push({
        content: `Le match ${match.homeTeam} – ${match.awayTeam} (${match.league}) s'est terminé sur le score de ${match.homeScore}-${match.awayScore}.`,
        type: "statistic",
        confidence: "verified",
        sources: ["internal:sports_cache"],
      });
    } else {
      // Match à venir
      facts.push({
        content: `${match.homeTeam} affronte ${match.awayTeam} en ${match.league} le ${dateStr}.`,
        type: "event",
        confidence: "verified",
        sources: ["internal:sports_cache"],
      });
    }
  }

  return facts;
}

function buildSportsSummary(sportsResult: {
  intent: string;
  matches: any[];
  teamName?: string;
  competition?: string;
  date?: Date;
}): string {
  const { intent, matches, teamName, competition, date } = sportsResult;
  if (!matches || matches.length === 0) {
    return "Aucun match correspondant trouvé pour cette requête dans les 7 prochains jours.";
  }

  const dateStr = date
    ? date.toLocaleDateString("fr-FR", { day: "2-digit", month: "2-digit" })
    : null;

  if (intent === "today_matches") {
    return `Il y a ${matches.length} match(s) de football aujourd'hui${dateStr ? ` (${dateStr})` : ""} dans les ligues suivies (L1, PL, Liga, Serie A, Bundesliga, C1, C3).`;
  }

  if (intent === "team_matches" && teamName) {
    return `J'ai trouvé ${matches.length} match(s) pour ${teamName}${dateStr ? ` le ${dateStr}` : ""} avec les cotes et les stats disponibles.`;
  }

  if (intent === "competition_matches" && competition) {
    return `J'ai trouvé ${matches.length} match(s) de ${competition} dans les prochains jours, avec les cotes et l'intérêt pour les paris.`;
  }

  if (intent === "week_matches") {
    return `Il y a ${matches.length} match(s) importants cette semaine dans les ligues majeures (L1, PL, Liga, Serie A, Bundesliga, C1, C3).`;
  }

  return `J'ai trouvé ${matches.length} match(s) correspondant à ta requête.`;
}
