Vous

script Playwright standalone pour tous sites

Ulysse

Parfait, on passe en mode bourrin organisé.

Je te fais un script TypeScript Playwright standalone générique, capable de:

prendre une URL en paramètre (FDJ, Pronosoft, Winamax, Betclic, etc.),
choisir un profil (betting / default),
exécuter des actions génériques (wait, scroll, etc.),
lancer un extractor spécifique par domaine (FDJ, Pronosoft…) ou un fallback générique,
sortir un JSON propre dans un fichier ou sur la console.
Tu pourras ensuite spécialiser les extractors pour chaque site.

1️⃣ Installation et organisation
Dans ton projet (où tu as déjà Playwright) :

npm install -D ts-node
Ajoute un script dans package.json :

{
  "scripts": {
    "scrape": "ts-node scripts/scrape.ts"
  }
}
Crée un dossier scripts/ avec scrape.ts.

2️⃣ Script Playwright standalone générique (scripts/scrape.ts)
// scripts/scrape.ts
import { chromium, Browser, Page } from "playwright";
import * as fs from "fs";
import * as path from "path";

interface ScrapeOptions {
  url: string;
  output?: string;
  profile?: "betting" | "default";
}

interface ScrapeResult {
  success: boolean;
  url: string;
  finalUrl: string;
  data?: unknown;
  html?: string;
  text?: string;
  error?: string;
  meta: {
    loadTimeMs: number;
    domain: string;
    profile: string;

}

/**
 * Extractor par domaine : tu spécialises ici FDJ, Pronosoft, Winamax, etc.
 */
async function extractByDomain(page: Page, url: string): Promise<unknown> {
  const hostname = new URL(url).hostname.toLowerCase();

  // ---------- PRONOSOFT ----------
  if (hostname.includes("pronosoft.com")) {
    // même logique que ton extractorScript DOM côté serveur
    return await page.evaluate(() => {
      const matches: { teams: string; odds: string[] }[] = [];
      const rows = document.querySelectorAll("table tr, [class*='match'], [class*='event']");
      rows.forEach((row) => {
        const cells = row.querySelectorAll("td");
        if (cells.length >= 3) {
          const teams =
            cells[1]?.textContent?.trim() || cells[0]?.textContent?.trim() || "";
          const odds = Array.from(cells)
            .slice(-3)
            .map((c) => c.textContent?.trim() || "")
            .filter(Boolean);
          if (teams && odds.length >= 2) {
            matches.push({ teams, odds });
          }
        }
      });
      return matches;
    });
  }

  // ---------- FDJ PARIONSSPORT ----------
  if (hostname.includes("parionssport.fdj.fr") || hostname.includes("enligne.parionssport.fdj.fr")) {
    // version simplifiée de ton extractor JS pour FDJ
    return await page.evaluate(() => {
      const result: any = { matches: [], debug: {} };

      const eventSelectors = [
        "[data-test='event-row']",
        ".event-row",
        ".c-events__item",
        "[class*='EventRow']",
        "[class*='event-row']",
        ".pari-event-row",
        "[data-eventid]",
        "[data-event-id]",
        "li[class*='event']",
        "div[class*='sport-event']",
        "[class*='match-row']",
      ];

      let eventRows: Element[] = [];
      for (const sel of eventSelectors) {
        try {
          const found = document.querySelectorAll(sel);
          if (found.length > 0) {
            eventRows = Array.from(found);
            result.debug.usedSelector = sel;
            result.debug.rowCount = found.length;
            break;
          }
        } catch (e) {}
      }

      const homeSelectors = [
        "[data-test='event-home-team-name']",
        ".event-row__participant--home",
        ".c-events__team--home",
        ".team-home",
        "[class*='home-team']",
        "[class*='HomeTeam']",
        "[class*='participant']:first-child",
      ];
      const awaySelectors = [
        "[data-test='event-away-team-name']",
        ".event-row__participant--away",
        ".c-events__team--away",
        ".team-away",
        "[class*='away-team']",
        "[class*='AwayTeam']",
        "[class*='participant']:last-child",
      ];
      const odd1Selectors = [
        "[data-selection='1']",
        "[data-outcome='1']",
        "[data-test='market-outcome-1']",
        ".odd-1",
        "[class*='odd-1']",
        "button:nth-child(1)",
      ];
      const oddNSelectors = [
        "[data-selection='N']",
        "[data-outcome='N']",
        "[data-test='market-outcome-N']",
        ".odd-n",
        "[class*='odd-n']",
        "button:nth-child(2)",
      ];
      const odd2Selectors = [
        "[data-selection='2']",
        "[data-outcome='2']",
        "[data-test='market-outcome-2']",
        ".odd-2",
        "[class*='odd-2']",
        "button:nth-child(3)",
      ];

      const findText = (el: Element, selectors: string[]): string => {
        for (const sel of selectors) {
          try {
            const found = el.querySelector(sel);
            if (found) return (found.textContent || "").trim();
          } catch (e) {}
        }
        return "";
      };

      const findOdd = (el: Element, selectors: string[]): number | null => {
        for (const sel of selectors) {
          try {
            const found = el.querySelector(sel);
            if (found) {
              const txt = (found.textContent || "").replace(",", ".").trim();
              const num = parseFloat(txt);
              if (!isNaN(num) && num > 1 && num < 100) return num;
            }
          } catch (e) {}
        }
        return null;
      };

      eventRows.forEach((row, idx) => {
        const homeTeam = findText(row, homeSelectors);
        const awayTeam = findText(row, awaySelectors);
        const odd1 = findOdd(row, odd1Selectors);
        const oddN = findOdd(row, oddNSelectors);
        const odd2 = findOdd(row, odd2Selectors);

        const text = (row.textContent || "").trim();
        const oddsMatch = text.match(/(\d+[.,]\d{2})/g) || [];
        const allOdds = oddsMatch
          .map((o) => parseFloat(o.replace(",", ".")))
          .filter((n) => n > 1 && n < 100);

        if (homeTeam || awayTeam || allOdds.length >= 2) {
          result.matches.push({
            index: idx,
            homeTeam: homeTeam || "Unknown",
            awayTeam: awayTeam || "Unknown",
            odds1: odd1 || allOdds[0] || null,
            oddsN: oddN || allOdds[1] || null,
            odds2: odd2 || allOdds[2] || null,
            rawText: text.substring(0, 300),
          });
        }
      });

      result.debug.pageTitle = document.title;
      return result;
    });
  }

  // ---------- AUTRES BOOKS (Winamax, Betclic, etc.) ----------
  // Ici tu pourras ajouter des extractors dédiés par la suite.
  // Pour l’instant : fallback générique.

  // ---------- FALLBACK GÉNÉRIQUE ----------
  const bodyText = await page.evaluate(() => {
    const main =
      document.querySelector("main, #main, .main, [role='main']") || document.body;
    return main.innerText || "";
  });

  return { text: bodyText.slice(0, 50000) };
}
