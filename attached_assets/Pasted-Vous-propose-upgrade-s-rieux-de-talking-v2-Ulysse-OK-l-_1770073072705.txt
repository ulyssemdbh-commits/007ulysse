2. Intégration FOOT sérieuse pour la voix
Idée
Pour tout ce qui est foot, /talking-v2 doit être branché en dur sur footdatas + cache Djedou Pronos + MatchEnDirect.

Upgrade
a) Intent + slot filling minimal

Tu peux faire un petit detectFootballIntent():

Exemples:

“qui est 3e en Ligue 1”

Tu passes le texte à un mini-LMM ou un prompt interne:

type FootballIntent =
  | { type: "last_match"; team: string }
  | { type: "next_match"; team: string }
  | { type: "ranking"; league: string }
  | { type: "topscorers"; league: string }
  | null;
b) Pipeline concret

async function answerFromFootballStack(userText: string) {
  const intent = detectFootballIntent(userText);
  if (!intent) return null; // laisse le LLM répondre normalement

  switch (intent.type) {
    case "last_match":
      return answerLastMatch(intent.team);
    case "next_match":
      return answerNextMatch(intent.team);
    case "ranking":
      return answerRanking(intent.league);
    case "topscorers":
      return answerTopscorers(intent.league);
  }
}
c) Exemple: dernier match OM

async function answerLastMatch(teamName: string) {
  // 1. DB footdatas
  const match = await footdatas.getLastMatchForTeam(teamName);
  if (match) return formatLastMatchVoice(match);

  // 2. Cache prédictions / cache matches
  const cached = await sportsCache.getLastMatch(teamName);
  if (cached) return formatLastMatchVoice(cached);

  // 3. Fallback matchendirect
  const live = await queryMatchEndirect(teamName);
  if (live) return formatLastMatchVoice(live);

  return "Je ne trouve pas de dernier match fiable pour " + teamName +
         ". Soit ils n'ont pas joué récemment, soit mes données sont incomplètes.";
}
