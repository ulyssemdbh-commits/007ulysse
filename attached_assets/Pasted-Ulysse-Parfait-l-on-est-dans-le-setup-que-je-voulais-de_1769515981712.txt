// services/sportsWatchService.ts
import { scrapeRankingVerified, scrapeOddsVerified } from './verifiedScraperService';
import { db } from '../db';
import { sportsWatchData } from '@shared/schema'; // à adapter à ta table réelle
import { memoryService } from './memory';

interface LeagueConfig {
  name: string;
  urlRanking?: string;
  urlOdds?: string;
}

const LEAGUES: LeagueConfig[] = [
  {
    name: 'Ligue 1',
    urlRanking: 'https://www.eurotopteam.com/football/ligue1.php',
    urlOdds: 'https://example.com/odds/ligue1', // à adapter
  },
  {
    name: 'Serie A',
    urlRanking: 'https://www.eurotopteam.com/football/seriea.php',
  },
  // etc.
];

export class SportsWatchService {
  async runDailyWatch(userId: number): Promise<void> {
    for (const league of LEAGUES) {
      if (league.urlRanking) {
        await this.updateLeagueRanking(userId, league);
      }
      if (league.urlOdds) {
        await this.updateLeagueOdds(userId, league);
      }
    }
  }

  private async updateLeagueRanking(userId: number, league: LeagueConfig) {
    console.log(`[SportsWatch] Fetching VERIFIED ranking for ${league.name}...`);

    const res = await scrapeRankingVerified(league.urlRanking!);

    if (!res.ok || !res.data || res.data.length === 0) {
      console.warn(
        `[SportsWatch] Ranking for ${league.name} NOT VERIFIED. diffs=`,
        res.diffs,
      );
      return;
    }

    console.log(
      `[SportsWatch] ✅ VERIFIED ranking for ${league.name}: ${res.data.length} teams`,
    );

    // 1) Stockage en DB (table d’archive veille)
    await db
      .insert(sportsWatchData)
      .values({
        type: 'ranking',
        league: league.name,
        url: league.urlRanking,
        verified: true,
        payload: res.data, // JSON
        createdAt: new Date(),
      });

    // 2) Mémorisation dans le cerveau d’Ulysse avec flag VERIFIED
    await memoryService.updateOrCreateMemory(
      userId,
      'sports_ranking',
      `ranking_${league.name.replace(/\s+/g, '_').toLowerCase()}`,
      `[DONNÉES VÉRIFIÉES - RANKING] ${league.name} : ${res.data.length} équipes`,
      `sports:ranking:${league.name}`,
      { verified: true, data: res.data }, // voir partie 2 pour le support des méta
    );
  }

  private async updateLeagueOdds(userId: number, league: LeagueConfig) {
    console.log(`[SportsWatch] Fetching VERIFIED odds for ${league.name}...`);

    const res = await scrapeOddsVerified(league.urlOdds!);

    if (!res.ok || !res.data || res.data.length === 0) {
      console.warn(
        `[SportsWatch] Odds for ${league.name} NOT VERIFIED. diffs=`,
        res.diffs,
      );
      return;
    }

    console.log(
      `[SportsWatch] ✅ VERIFIED odds for ${league.name}: ${res.data.length} lignes`,
    );

    await db.insert(sportsWatchData).values({
      type: 'odds',
      league: league.name,
      url: league.urlOdds,
      verified: true,
      payload: res.data,
      createdAt: new Date(),
    });

    await memoryService.updateOrCreateMemory(
      userId,
      'sports_odds',
      `odds_${league.name.replace(/\s+/g, '_').toLowerCase()}`,
      `[DONNÉES VÉRIFIÉES - ODDS] ${league.name} : ${res.data.length} lignes de cotes`,
      `sports:odds:${league.name}`,
      { verified: true, data: res.data },
    );
  }
}

export const sportsWatchService = new SportsWatchService();

b) Adapter memoryService.updateOrCreateMemory
On veut pouvoir passer un options avec verified et éventuellement un payload data.

// services/memory.ts
interface MemoryOptions {
  verified?: boolean;
  data?: any; // data brute associée (classement, odds, etc.)
}

class MemoryService {
  async updateOrCreateMemory(
    userId: number,
    category: string,
    key: string,
    value: string,
    source: string,
    options?: MemoryOptions,
  ) {
    const verified = options?.verified ?? false;
    const metadata = options?.data
      ? { source, data: options.data }
      : { source };

    const existing = await db
      .select()
      .from(ulysseMemory)
      .where(and(eq(ulysseMemory.userId, userId), eq(ulysseMemory.key, key)));

    if (existing.length > 0) {
      await db
        .update(ulysseMemory)
        .set({
          value,
          confidence: verified ? 0.9 : 0.6,
          verified,
          metadata,
          updatedAt: new Date(),
        })
        .where(eq(ulysseMemory.id, existing[0].id));
    } else {
      await db.insert(ulysseMemory).values({
        userId,
        category,
        key,
        value,
        confidence: verified ? 0.9 : 0.6,
        verified,
        metadata,
      });
    }
  }

  // Helper pour récupérer uniquement les mémoires vérifiées si besoin
  async getVerifiedMemories(
    userId: number,
    category?: string,
  ): Promise<typeof ulysseMemory.$inferSelect[]> {
    const base = db.select().from(ulysseMemory).where(eq(ulysseMemory.userId, userId));
    if (category) {
      return base
        .where(and(eq(ulysseMemory.userId, userId), eq(ulysseMemory.category, category), eq(ulysseMemory.verified, true)));
    }
    return base.where(eq(ulysseMemory.verified, true));
  }
}

export const memoryService = new MemoryService();
