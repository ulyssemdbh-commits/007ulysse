1. Vision globale du pipeline
Flow idéal :

Entrée (texte/voix/discord) → HearingHub
HearingHub normalise → envoie au BrainHub
BrainHub:
détecte le contexte (persona, speaker, domaine),
enrichit avec mémoire / état,
décide: réponse simple / outils / actions.
Actions → ActionHub (tool calls)
Réponse → VoiceOutputHub (voix + texte + UI)
BrainHub met à jour mémoire + stats.

D. BrainHub – chef d’orchestre
Objectif: Tout passe ici. Il:

reçoit les HearingEvents,
fait intent routing (VoiceBrain, Action-First),
décide actions + format de réponse,
parle via VoiceOutputHub,
agit via ActionHub,
met à jour la mémoire.
D1. Interface & squelette
// server/services/brain/brainHub.ts
import { hearingHub } from "../sensory/hearingHub";
import { voiceOutputHub } from "../sensory/voiceOutputHub";
import { actionHub } from "../sensory/actionHub";
import { detectWorkflow } from "../actionFirst/ulysseBehaviorRules";
import { voiceIntentRouter } from "../voice/voiceIntentRouter";
import { voiceContextMemory } from "../voice/voiceContextMemory";

export class BrainHub {
  async processInput(event: HearingEvent) {
    // 1) Construire un contexte cognitif initial
    const cognitiveContext = await this.buildCognitiveContext(event);

    // 2) Intent routing (VoiceBrain + Action-First)
    const domainIntent = await voiceIntentRouter.route(event.text, cognitiveContext);

    // 3) Selon intent: 
    //    - domain handler (foot, sugu, etc.)
    //    - ou workflow Action-First (email, task, search...)
    const { plannedActions, responseDraft, uiCard } =
      await this.planResponse(domainIntent, cognitiveContext);

    // 4) Exécuter les actions (si il y en a)
    let actionResults: ActionResultSummary[] = [];
    if (plannedActions.length > 0) {
      actionResults = await actionHub.execute(plannedActions, cognitiveContext.actionContext);
    }

    // 5) Synthétiser la réponse finale (texte + TTS + UI) 
    const outputPlan: OutputPlan = this.buildOutputPlan({
      event,
      domainIntent,
      responseDraft,
      actionResults,
      uiCard,
      cognitiveContext
    });

    // 6) Envoyer via VoiceOutputHub
    await voiceOutputHub.dispatch(outputPlan, { source: event.source, userId: event.userId, meta: event.meta });

    // 7) Mettre à jour la mémoire contextuelle
    await voiceContextMemory.update(event, domainIntent, { actionResults });

    // 8) Retourner quelque chose au caller (utile pour HTTP)
    return outputPlan;
  }

  private async buildCognitiveContext(event: HearingEvent) {
    // persona, speaker, mémoire courte (5 sujets), capacités, état système...
    return {
      persona: event.persona,
      speakerId: event.speakerId,
      // etc.
    };
  }

  private async planResponse(domainIntent: any, context: any) {
    // si intent = football.* → appeler handler foot
    // si intent = sugu.* → handler sugu
    // si workflow Action-First détecté (email/task/domotique/...) → plannedActions
    // sinon → Core Engine (LLM) pour réponse conversationnelle simple
    return {
      plannedActions: [],
      responseDraft: "Texte à dire",
      uiCard: null
    };
  }

  private buildOutputPlan(args: {
    event: HearingEvent;
    domainIntent: any;
    responseDraft: string;
    actionResults: ActionResultSummary[];
    uiCard: any;
    cognitiveContext: any;
  }): OutputPlan {
    const channels: OutputPlan["channels"] = [];

    if (args.event.source === "web_chat") channels.push("web_chat");
    if (args.event.source === "talking_v3") channels.push("talking_v3");
    if (args.event.source === "discord") channels.push("discord_text");

    return {
      channels,
      text: args.responseDraft,
      tts: args.event.isVoice ? { enabled: true } : { enabled: false },
      ui: args.uiCard
    };
  }
}

export const brainHub = new BrainHub();
