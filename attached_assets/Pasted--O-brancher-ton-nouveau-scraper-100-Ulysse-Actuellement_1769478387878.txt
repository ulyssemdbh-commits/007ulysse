import { crawlAndScrape } from "./scraperService";
Exemple d’adaptation pour 1 URL
private async executeUrlFetchTask(
  userId: number,
  homework: UlysseHomework,
  url: string,
  personaName: string = "Ulysse"
): Promise<{ summary: string; artifacts: any }> {
  try {
    console.log(`[HomeworkExecution] Crawling URL via Super Ulysse Scraper: ${url}`);

    const scrapeResult = await crawlAndScrape({
      url,
      mode: "http",          // ou "auto", mais tout reste HTTP-only côté scraper
      depth: 0,
      maxPages: 1,
      maxConcurrency: 1,
      linkScope: "none",
      extract: {
        text: true,
        links: false,
        metadata: true,
      },
    });

    if (!scrapeResult.pages.length || scrapeResult.pages[0].status !== 200) {
      const errorMsg =
        scrapeResult.pages[0]?.error ||
        `HTTP ${scrapeResult.pages[0]?.status ?? 0}`;
      return {
        summary: `Impossible d'accéder au site ${url}. ${errorMsg}`,
        artifacts: { url, error: errorMsg, processed: false },
      };
    }

    const page = scrapeResult.pages[0];
    const title = page.metadata?.title || url;
    const text = page.text || "";

    // Si pas d'IA dispo → fallback brut
    if (!canMakeCall("combined")) {
      return {
        summary: `Contenu extrait de ${url}:\n\n**${title}**\n\n${text.substring(
          0,
          1000,
        )}...`,
        artifacts: {
          url,
          title,
          contentLength: text.length,
          processed: true,
        },
      };
    }

    // META-PROMPTING: prompt optimisé
    const optimizedPrompt = await this.generateOptimizedPrompt(
      homework,
      "url_fetch",
      personaName,
    );

    const analysisResponse = await withRateLimit("combined", () =>
      openai.chat.completions.create({
        model: "gpt-4o-mini",
        messages: [
          { role: "system", content: optimizedPrompt },
          {
            role: "user",
            content: `Tâche: ${homework.title}\n${homework.description ||
              ""}\n\nURL: ${url}\nTitre: ${title}\n\nCONTENU:\n${text}`,
          },
        ],
        temperature: 0.3,
        max_tokens: 800,
      }),
      0,
    );

    return {
      summary:
        analysisResponse.choices[0].message.content ||
        `Contenu récupéré de ${url}`,
      artifacts: {
        url,
        title,
        contentLength: text.length,
        processed: true,
      },
    };
  } catch (error) {
    console.error("[HomeworkExecution] URL fetch error:", error);
    return {
      summary: `Erreur lors de l'accès à ${url}: ${
        error instanceof Error ? error.message : String(error)
      }`,
      artifacts: { url, error: String(error), processed: false },
    };
  }
}
Pour la version multi-URLs, même idée:

tu remplaces les appels à crawl(...) par crawlAndScrape(...) sur chaque URL,
tu utilises page.text / page.metadata.title pour construire combinedContent.
