Tasks.tsx – V2 optimale
import { PageContainer } from "@/components/layout/PageContainer";
import { CreateTaskDialog } from "@/components/CreateTaskDialog";
import {
  useTasks,
  useUpdateTask,
  useDeleteTask,
} from "@/hooks/use-tasks";
import {
  Card,
  CardContent,
  CardHeader,
  CardTitle,
} from "@/components/ui/card";
import { StatusBadge } from "@/components/ui/StatusBadge";
import { Skeleton } from "@/components/ui/skeleton";
import { Button } from "@/components/ui/button";
import {
  Check,
  Clock,
  ListTodo,
  Trash2,
  AlertTriangle,
  Calendar as CalendarIcon,
  Filter,
} from "lucide-react";
import { motion, AnimatePresence } from "framer-motion";
import { cn } from "@/lib/utils";
import { useMemo, useState } from "react";
import { format, isBefore, isToday, isAfter, startOfDay } from "date-fns";
import { fr } from "date-fns/locale";
import {
  Select,
  SelectContent,
  SelectItem,
  SelectTrigger,
  SelectValue,
} from "@/components/ui/select";
import {
  Tooltip,
  TooltipContent,
  TooltipProvider,
  TooltipTrigger,
} from "@/components/ui/tooltip";
import { useToast } from "@/components/ui/use-toast";
import type { Task } from "@shared/schema";

const columns = [
  { id: "todo", title: "To Do", icon: ListTodo, color: "text-yellow-500" },
  { id: "in_progress", title: "In Progress", icon: Clock, color: "text-blue-500" },
  { id: "done", title: "Done", icon: Check, color: "text-green-500" },
] as const;

type StatusId = (typeof columns)[number]["id"];

type Priority = "low" | "medium" | "high";

const priorityLabel: Record<Priority, string> = {
  low: "Basse",
  medium: "Normale",
  high: "Haute",
};

const priorityOrder: Record<Priority, number> = {
  high: 0,
  medium: 1,
  low: 2,
};

type DateFilter = "all" | "today" | "overdue" | "upcoming";

export default function Tasks() {
  const { data: tasks, isLoading } = useTasks();
  const updateTask = useUpdateTask();
  const deleteTask = useDeleteTask();
  const { toast } = useToast();

  const [statusFilter, setStatusFilter] = useState<StatusId | "all">("all");
  const [dateFilter, setDateFilter] = useState<DateFilter>("all");
  const [priorityFilter, setPriorityFilter] = useState<Priority | "all">("all");

  const today = startOfDay(new Date());

  const handleStatusChange = (id: number, newStatus: StatusId) => {
    const previousTasks = tasks;

    updateTask.mutate(
      { id, status: newStatus },
      {
        onError: (error) => {
          console.error(error);
          toast({
            variant: "destructive",
            title: "Échec de la mise à jour",
            description: "Impossible de changer le statut de la tâche.",
          });
          // Optionnel: on pourrait remettre à jour le cache avec previousTasks
        },
        onSuccess: () => {
          toast({
            title: "Tâche mise à jour",
            description: "Le statut a été modifié avec succès.",
          });
        },
      }
    );
  };

  const handleDelete = (id: number) => {
    if (!confirm("Tu es sûr de vouloir supprimer cette tâche ?")) return;

    deleteTask.mutate(id, {
      onError: (error) => {
        console.error(error);
        toast({
          variant: "destructive",
          title: "Échec de la suppression",
          description: "Impossible de supprimer la tâche.",
        });
      },
      onSuccess: () => {
        toast({
          title: "Tâche supprimée",
          description: "La tâche a été supprimée.",
        });
      },
    });
  };

  const filteredTasks = useMemo(() => {
    if (!tasks) return [];

    return tasks
      .filter((task) => {
        // Filtre statut global (barre de filtres)
        if (statusFilter !== "all" && task.status !== statusFilter) return false;

        // Filtre priorité
        if (priorityFilter !== "all") {
          const p = (task.priority || "medium") as Priority;
          if (p !== priorityFilter) return false;
        }

        // Filtre date
        if (dateFilter !== "all") {
          if (!task.dueDate) {
            // si pas de dueDate et filtre sur date → on peut décider de les exclure
            return false;
          }

          const due = startOfDay(new Date(task.dueDate));

          if (dateFilter === "today") {
            return isToday(due);
          }

          if (dateFilter === "overdue") {
            return isBefore(due, today);
          }

          if (dateFilter === "upcoming") {
            return isAfter(due, today);
          }
        }

        return true;
      })
      .sort((a, b) => {
        // tri global: d'abord par statut, puis par priorité, puis dueDate
        const statusIndexA = columns.findIndex((c) => c.id === a.status);
        const statusIndexB = columns.findIndex((c) => c.id === b.status);
        if (statusIndexA !== statusIndexB) return statusIndexA - statusIndexB;

        const priorityA = (a.priority || "medium") as Priority;
        const priorityB = (b.priority || "medium") as Priority;
        if (priorityOrder[priorityA] !== priorityOrder[priorityB]) {
          return priorityOrder[priorityA] - priorityOrder[priorityB];
        }

        if (a.dueDate && b.dueDate) {
          return new Date(a.dueDate).getTime() - new Date(b.dueDate).getTime();
        }

        return 0;
      });
  }, [tasks, statusFilter, dateFilter, priorityFilter, today]);

  const tasksByColumn = useMemo(() => {
    const map: Record<StatusId, Task[]> = {
      todo: [],
      in_progress: [],
      done: [],
    };

    for (const t of filteredTasks) {
      if (t.status === "todo" || t.status === "in_progress" || t.status === "done") {
        map[t.status].push(t);
      }
    }

    // Tri par priorité et dueDate à l’intérieur de chaque colonne
    (Object.keys(map) as StatusId[]).forEach((status) => {
      map[status].sort((a, b) => {
        const pa = (a.priority || "medium") as Priority;
        const pb = (b.priority || "medium") as Priority;
        if (priorityOrder[pa] !== priorityOrder[pb]) {
          return priorityOrder[pa] - priorityOrder[pb];
        }
        if (a.dueDate && b.dueDate) {
          return new Date(a.dueDate).getTime() - new Date(b.dueDate).getTime();
        }
        return 0;
      });
    });

    return map;
  }, [filteredTasks]);

  const isMutating = updateTask.isPending || deleteTask.isPending;

  if (isLoading) {
    return (
      <PageContainer title="Tasks">
        <div className="flex items-center justify-between mb-4">
          <div className="h-8 w-40 rounded bg-muted animate-pulse" />
          <div className="h-9 w-32 rounded bg-muted animate-pulse" />
        </div>
        <div className="grid md:grid-cols-3 gap-6">
          {[1, 2, 3].map((i) => (
            <Skeleton key={i} className="h-[600px] rounded-xl" />
          ))}
        </div>
      </PageContainer>
    );
  }

  return (
    <PageContainer
      title="Tasks"
      action={<CreateTaskDialog />}
    >
      {/* Barre de filtres globale */}
      <div className="mb-4 flex flex-col md:flex-row gap-3 items-start md:items-center justify-between">
        <div className="flex items-center gap-2 text-sm text-muted-foreground">
          <Filter className="w-4 h-4" />
          <span>Filtrer les tâches</span>
        </div>

        <div className="flex flex-wrap gap-2">
          {/* Filtre statut */}
          <Select
            value={statusFilter}
            onValueChange={(v) => setStatusFilter(v as StatusId | "all")}
          >
            <SelectTrigger className="w-[140px]">
              <SelectValue placeholder="Statut" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">Tous les statuts</SelectItem>
              {columns.map((c) => (
                <SelectItem key={c.id} value={c.id}>
                  {c.title}
                </SelectItem>
              ))}
            </SelectContent>
          </Select>

          {/* Filtre date */}
          <Select
            value={dateFilter}
            onValueChange={(v) => setDateFilter(v as DateFilter)}
          >
            <SelectTrigger className="w-[160px]">
              <SelectValue placeholder="Date" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">Toutes les dates</SelectItem>
              <SelectItem value="today">Aujourd&apos;hui</SelectItem>
              <SelectItem value="overdue">En retard</SelectItem>
              <SelectItem value="upcoming">À venir</SelectItem>
            </SelectContent>
          </Select>

          {/* Filtre priorité */}
          <Select
            value={priorityFilter}
            onValueChange={(v) => setPriorityFilter(v as Priority | "all")}
          >
            <SelectTrigger className="w-[150px]">
              <SelectValue placeholder="Priorité" />
            </SelectTrigger>
            <SelectContent>
              <SelectItem value="all">Toutes priorités</SelectItem>
              <SelectItem value="high">Haute</SelectItem>
              <SelectItem value="medium">Normale</SelectItem>
              <SelectItem value="low">Basse</SelectItem>
            </SelectContent>
          </Select>
        </div>
      </div>

      <TooltipProvider>
        <div className="grid md:grid-cols-3 gap-6 h-full overflow-hidden">
          {columns.map((col) => {
            const colTasks = tasksByColumn[col.id];

            return (
              <div
                key={col.id}
                className={cn(
                  "flex flex-col h-[calc(100vh-210px)] bg-secondary/20 rounded-xl border border-border/50 p-4"
                )}
              >
                <div className="flex items-center justify-between mb-4">
                  <div className="flex items-center gap-2">
                    <col.icon className={cn("w-5 h-5", col.color)} />
                    <h3 className="font-semibold text-lg">{col.title}</h3>
                  </div>
                  <span className="bg-secondary px-2 py-1 rounded text-xs font-medium text-muted-foreground">
                    {colTasks.length}
                  </span>
                </div>

                <div className="space-y-3 overflow-y-auto flex-1 pr-2 scrollbar-thin scrollbar-thumb-border">
                  <AnimatePresence>
                    {colTasks.map((task) => {
                      const priority = (task.priority || "medium") as Priority;
                      const hasDueDate = !!task.dueDate;
                      const dueDate = hasDueDate
                        ? new Date(task.dueDate as string)
                        : null;

                      const isOverdue =
                        hasDueDate && isBefore(startOfDay(dueDate!), today);
                      const isTodayDue =
                        hasDueDate && isToday(startOfDay(dueDate!));

                      return (
                        <motion.div
                          key={task.id}
                          layout
                          initial={{ opacity: 0, scale: 0.95, y: 8 }}
                          animate={{ opacity: 1, scale: 1, y: 0 }}
                          exit={{ opacity: 0, scale: 0.95, y: -8 }}
                          transition={{ duration: 0.15 }}
                        >
                          <Card
                            className={cn(
                              "bg-card border-border shadow-sm hover:shadow-md hover:border-primary/30 transition-all group relative",
                              isOverdue &&
                                "border-destructive/60 bg-destructive/5",
                              isTodayDue && "border-amber-400/70 bg-amber-50/5"
                            )}
                          >
                            <CardHeader className="p-4 pb-2">
                              <div className="flex justify-between items-start gap-2">
                                <div className="flex flex-col gap-1">
                                  <CardTitle className="text-sm font-medium leading-tight">
                                    {task.title}
                                  </CardTitle>

                                  {/* Due date + status visuel */}
                                  {hasDueDate && (
                                    <div className="flex items-center gap-1 text-[11px] text-muted-foreground">
                                      <CalendarIcon className="w-3 h-3" />
                                      <span>
                                        {format(dueDate!, "dd MMM yyyy", {
                                          locale: fr,
                                        })}
                                      </span>
                                      {isOverdue && (
                                        <span className="inline-flex items-center gap-1 text-destructive font-medium ml-1">
                                          <AlertTriangle className="w-3 h-3" />
                                          En retard
                                        </span>
                                      )}
                                      {isTodayDue && !isOverdue && (
                                        <span className="inline-flex items-center gap-1 text-amber-500 font-medium ml-1">
                                          Aujourd&apos;hui
                                        </span>
                                      )}
                                    </div>
                                  )}
                                </div>
                                <Tooltip>
                                  <TooltipTrigger asChild>
                                    <div>
                                      <StatusBadge status={priority} />
                                    </div>
                                  </TooltipTrigger>
                                  <TooltipContent side="left">
                                    Priorité : {priorityLabel[priority]}
                                  </TooltipContent>
                                </Tooltip>
                              </div>
                            </CardHeader>

                            <CardContent className="p-4 pt-2">
                              <p className="text-xs text-muted-foreground line-clamp-3 mb-3">
                                {task.description || "Pas de description"}
                              </p>

                              <div className="flex justify-between items-center opacity-0 group-hover:opacity-100 transition-opacity">
                                <div className="flex gap-1">
                                  {col.id !== "todo" && (
                                    <Tooltip>
                                      <TooltipTrigger asChild>
                                        <Button
                                          variant="ghost"
                                          size="icon"
                                          className="h-6 w-6"
                                          onClick={() =>
                                            handleStatusChange(task.id, "todo")
                                          }
                                          disabled={isMutating}
                                        >
                                          <ListTodo className="w-3 h-3" />
                                        </Button>
                                      </TooltipTrigger>
                                      <TooltipContent>
                                        Repasser en &quot;To Do&quot;
                                      </TooltipContent>
                                    </Tooltip>
                                  )}
                                  {col.id !== "in_progress" && (
                                    <Tooltip>
                                      <TooltipTrigger asChild>
                                        <Button
                                          variant="ghost"
                                          size="icon"
                                          className="h-6 w-6"
                                          onClick={() =>
                                            handleStatusChange(
                                              task.id,
                                              "in_progress"
                                            )
                                          }
                                          disabled={isMutating}
                                        >
                                          <Clock className="w-3 h-3" />
                                        </Button>
                                      </TooltipTrigger>
                                      <TooltipContent>
                                        Passer en &quot;In Progress&quot;
                                      </TooltipContent>
                                    </Tooltip>
                                  )}
                                  {col.id !== "done" && (
                                    <Tooltip>
                                      <TooltipTrigger asChild>
                                        <Button
                                          variant="ghost"
                                          size="icon"
                                          className="h-6 w-6"
                                          onClick={() =>
                                            handleStatusChange(task.id, "done")
                                          }
                                          disabled={isMutating}
                                        >
                                          <Check className="w-3 h-3" />
                                        </Button>
                                      </TooltipTrigger>
                                      <TooltipContent>
                                        Marquer comme terminée
                                      </TooltipContent>
                                    </Tooltip>
                                  )}
                                </div>

                                <Tooltip>
                                  <TooltipTrigger asChild>
                                    <Button
                                      variant="ghost"
                                      size="icon"
                                      className="h-6 w-6 text-destructive hover:text-destructive hover:bg-destructive/10"
                                      onClick={() => handleDelete(task.id)}
                                      disabled={deleteTask.isPending}
                                    >
                                      <Trash2 className="w-3 h-3" />
                                    </Button>
                                  </TooltipTrigger>
                                  <TooltipContent>
                                    Supprimer la tâche
                                  </TooltipContent>
                                </Tooltip>
                              </div>
                            </CardContent>
                          </Card>
                        </motion.div>
                      );
                    })}

                    {colTasks.length === 0 && (
                      <motion.div
                        layout
                        initial={{ opacity: 0 }}
                        animate={{ opacity: 0.7 }}
                        className="flex flex-col items-center justify-center py-8 text-muted-foreground/50 border-2 border-dashed border-border/30 rounded-lg text-xs"
                      >
                        <p>Aucune tâche</p>
                        <p className="mt-1 text-[11px]">
                          Ajuste les filtres ou crée une nouvelle tâche.
                        </p>
                      </motion.div>
                    )}
                  </AnimatePresence>
                </div>
              </div>
            );
          })}
        </div>
      </TooltipProvider>
    </PageContainer>
  );
}
Ce que cette V2 t’apporte concrètement
Pilotage sérieux

Filtres par statut, date (aujourd’hui / en retard / à venir), priorité.
Tri par statut → priorité → dueDate, à la fois globalement et par colonne.
Lisibilité et gestion du temps

Affichage de la dueDate + badges “En retard” / “Aujourd’hui”.
Les tâches critiques ressortent visuellement (bordures, couleurs).
Robustesse UX

useToast sur succès/erreur (update + delete).
Boutons désactivés pendant mutation (isMutating).
Animations propres avec AnimatePresence.
