3. Flux concret côté serveur (realtimeVoice.ts)
Pseudo-code très concret, adapté à ce que tu as :

// Paramètres réglables
const SILENCE_THRESHOLD = 0.0015; // à ajuster selon ton niveau audio
const SILENCE_DURATION_MS = 800;
const MAX_TURN_MS = 12000;
const MIN_WORDS_FOR_EOS = 3;

function isSilent(pcmChunk: Int16Array) {
  let sum = 0;
  for (let i = 0; i < pcmChunk.length; i++) {
    sum += pcmChunk[i] * pcmChunk[i];
  }
  const rms = Math.sqrt(sum / pcmChunk.length) / 32768;
  return rms < SILENCE_THRESHOLD;
}

async function handleIncomingAudioChunk(chunk: Buffer, call: TurnState) {
  const now = Date.now();
  call.lastAudioTime = now;

  const pcm = new Int16Array(chunk.buffer, chunk.byteOffset, chunk.byteLength / 2);

  // 1) Silence / parole
  if (!isSilent(pcm)) {
    call.isSpeaking = true;
    call.lastSpeechTime = now;
    // accumulate in buffer for Whisper...
  } else {
    // chunk silencieux
    if (call.isSpeaking && now - call.lastSpeechTime > SILENCE_DURATION_MS) {
      // On considère que l'utilisateur a fini de parler
      call.isSpeaking = false;
      await maybeEndTurn(call);
    }
  }
}

// Appelé après mise à jour du transcript (Whisper)
async function onTranscriptUpdate(newTranscript: string, call: TurnState) {
  call.currentTranscript = newTranscript;

  const words = newTranscript.trim().split(/\s+/);
  const endsWithPunct = /[.?!…]$/.test(newTranscript.trim());

  const now = Date.now();
  const timeSinceLastSpeech = now - call.lastSpeechTime;
  const timeSinceStart = now - call.turnStartTime;

  const enoughWords = words.length >= MIN_WORDS_FOR_EOS;

  const silenceLong = timeSinceLastSpeech > SILENCE_DURATION_MS;
  const forcedByTime = timeSinceStart > MAX_TURN_MS;

  if (
    (endsWithPunct && enoughWords && silenceLong) ||
    (silenceLong && enoughWords) ||
    forcedByTime
  ) {
    await endTurn(call);
  }
}

async function endTurn(call: TurnState) {
  const text = call.currentTranscript.trim();
  if (!text) return;

  // Reset turnStartTime pour le prochain tour
  call.turnStartTime = Date.now();

  // Ici on envoie à Ulysse / .org
  const result = await callUlysseChat({ message: text, source: 'talking', userId: call.userId });

  // TTS + actions
  await handleUlysseResult(result);
}

// turnDetection.ts
export type TurnDetectionConfig = {
  silenceThreshold: number;      // ex: 0.0015
  silenceDurationMs: number;     // ex: 800
  maxTurnMs: number;             // ex: 12000
  minWordsForEos: number;        // ex: 3
};

export type TurnState = {
  currentTranscript: string;
  turnStartTime: number;
  lastSpeechTime: number;
  isSpeaking: boolean;
};

export class TurnDetector {
  private config: TurnDetectionConfig;
  private state: TurnState;
  private onEndTurn: (finalTranscript: string) => void | Promise<void>;

  constructor(config: TurnDetectionConfig, onEndTurn: (text: string) => void | Promise<void>) {
    this.config = config;
    this.state = {
      currentTranscript: '',
      turnStartTime: Date.now(),
      lastSpeechTime: Date.now(),
      isSpeaking: false,
    };
    this.onEndTurn = onEndTurn;
  }

  handleAudioChunk(pcm: Int16Array) {
    const now = Date.now();

    if (!this.isSilent(pcm)) {
      this.state.isSpeaking = true;
      this.state.lastSpeechTime = now;
    } else {
      const sinceLastSpeech = now - this.state.lastSpeechTime;
      if (this.state.isSpeaking && sinceLastSpeech > this.config.silenceDurationMs) {
        this.state.isSpeaking = false;
        this.maybeEndTurn('silence');
      }
    }

    const turnDuration = now - this.state.turnStartTime;
    if (turnDuration > this.config.maxTurnMs) {
      this.maybeEndTurn('timeout');
    }
  }

  handleTranscriptUpdate(newTranscript: string) {
    this.state.currentTranscript = newTranscript;
    this.maybeEndTurn('text');
  }

  private async maybeEndTurn(reason: 'silence' | 'timeout' | 'text') {
    const text = this.state.currentTranscript.trim();
    if (!text) return;

    const words = text.split(/\s+/).filter(Boolean);
    const endsWithPunct = /[.?!…]$/.test(text);

    const now = Date.now();
    const sinceLastSpeech = now - this.state.lastSpeechTime;

    const enoughWords = words.length >= this.config.minWordsForEos;
    const silenceLong = sinceLastSpeech > this.config.silenceDurationMs;
    const timeout = now - this.state.turnStartTime > this.config.maxTurnMs;

    const eos =
      timeout ||
      (reason === 'silence' && silenceLong && enoughWords) ||
      (reason === 'text' && endsWithPunct && enoughWords && silenceLong);

    if (!eos) return;

    // reset pour prochain tour
    this.state.turnStartTime = Date.now();
    this.state.currentTranscript = '';

    await this.onEndTurn(text);
  }

  private isSilent(pcm: Int16Array) {
    let sum = 0;
    for (let i = 0; i < pcm.length; i++) {
      sum += pcm[i] * pcm[i];
    }
    const rms = Math.sqrt(sum / pcm.length) / 32768;
    return rms < this.config.silenceThreshold;
  }
}
