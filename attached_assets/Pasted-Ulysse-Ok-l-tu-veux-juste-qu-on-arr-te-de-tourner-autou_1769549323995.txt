1. Scraper générique (rappel rapide)
On réutilise l’orchestrateur que je t’ai donné :

// src/scraper.ts
import { HttpFetcher } from "./httpFetcher";
import { BrowserFetcher } from "./browserFetcher";
import { detectBlock } from "./blockDetection";

export interface ScrapeResult {
  url: string;
  html: string;
  from: "http" | "browser";
}

export class UniversalScraper {
  private http: HttpFetcher;
  private browser: BrowserFetcher;

  constructor() {
    this.http = new HttpFetcher();
    this.browser = new BrowserFetcher();
  }

  async scrape(url: string, opts?: { forceBrowser?: boolean; waitSelector?: string; }): Promise<ScrapeResult> {
    if (!opts?.forceBrowser) {
      const res = await this.http.fetch(url);
      const block = detectBlock(res.html, res.status);
      if (!block.blocked && res.status >= 200 && res.status < 400) {
        return { url: res.url, html: res.html, from: "http" };
      }
    }
    const bRes = await this.browser.fetch(url, { waitForSelector: opts?.waitSelector ?? "body" });
    const block2 = detectBlock(bRes.html, bRes.status);
    if (block2.blocked) {
      throw new Error(`Blocked by protection: ${block2.reason ?? "unknown"}`);
    }
    return { url: bRes.url, html: bRes.html, from: "browser" };
  }
}
2. Extraction spécifique ParionsSport Champions League
Sur ParionsSport, les matchs sont rendus en HTML avec une structure du genre (à adapter selon l’HTML réel, mais l’idée est là) :

<div class="event-row">
  <div class="team home">Manchester City</div>

  <button class="odd odd-1">1.20</button>
  <button class="odd odd-n">5.50</button>

</div>
On va utiliser cheerio pour parser.

// src/parionssportParser.ts
import * as cheerio from "cheerio";
import { MatchOdds } from "./models";

export function parseChampionsLeague(html: string): MatchOdds[] {
  const $ = cheerio.load(html);
  const results: MatchOdds[] = [];

  // À ajuster selon le DOM réel :
  // inspecte la page dans ton navigateur et adapte les sélecteurs ici
  const rows = $("[data-test='event-row'], .event-row, .pari-event-row");

  rows.each((_, el) => {
    const row = $(el);

    const homeTeam =
      row.find("[data-test='event-home-team-name'], .team-home, .event-row__participant--home").text().trim();
    const awayTeam =
      row.find("[data-test='event-away-team-name'], .team-away, .event-row__participant--away").text().trim();

    if (!homeTeam || !awayTeam) {
      return;
    }

    // Les cotes 1N2 – adapter les classes / data-attrs en fonction du HTML réel
    const odd1Text =
      row.find("[data-selection='1'], [data-outcome='1'], .odd-1").first().text().replace(",", ".").trim();
    const oddNText =
      row.find("[data-selection='N'], [data-outcome='N'], .odd-n").first().text().replace(",", ".").trim();
    const odd2Text =
      row.find("[data-selection='2'], [data-outcome='2'], .odd-2").first().text().replace(",", ".").trim();

    const odds1 = odd1Text ? Number(odd1Text) : null;
    const oddsN = oddNText ? Number(oddNText) : null;
    const odds2 = odd2Text ? Number(odd2Text) : null;

    results.push({
      competition: "Ligue des Champions",
      homeTeam,
      awayTeam,
      odds1: isNaN(Number(odds1)) ? null : odds1,
      oddsN: isNaN(Number(oddsN)) ? null : oddsN,
      odds2: isNaN(Number(odds2)) ? null : odds2
    });
  });

  return results;
}

const URL = "https://www.enligne.parionssport.fdj.fr/paris-football/coupes-d-europe/championsleague";

async function main() {
  const scraper = new UniversalScraper();

  console.log("Scraping:", URL);
  const res = await scraper.scrape(URL, {
