1️⃣ Objectif : “Scraping Vérifié” pour TOUT
On veut un pipeline générique :

Scraper une page → résultat brut (HTML)
Extraire des données structurées (avec une fonction d’extraction spécifique au cas d’usage)
Rescraper + réextraire (au moins une deuxième fois)
Comparer les deux extractions
Si c’est identique / cohérent → on marque le résultat comme VALIDÉ
Si c’est différent → on:
réessaie (3ᵉ tentative), ou
marque comme non fiable et on ne l’utilise pas comme “vérité”
Et ça doit marcher pour :

classements foot
listes de produits
articles d’actualités
stats, tableaux, etc.
Donc on a besoin de trois couches génériques :

fetchHtml(url) → HTML brut (ce que ton scraper fait déjà)
extractFn(html) => T → fonction d’extraction spécifique au type de donnée (classement, produits, article…)
compareFn(a: T, b: T) => Diff[] → fonction qui sait dire si deux résultats T sont équivalents ou pas
2️⃣ Interface générique pour “scrape vérifié”
Types génériques
export interface VerifiedScrapeResult<T> {
  ok: boolean;          // true = résultat validé
  data?: T;             // données validées
  attempts: number;     // nombre de tentatives
  diffs?: string[];     // différences trouvées entre tentatives
}

export interface VerifyOptions {
  maxAttempts?: number;   // default: 2
  delayMs?: number;       // default: 500ms entre tentatives
}
Fonction générique
// À mettre dans ton scraperService ou un module dédié
async function verifiedScrape<T>(
  url: string,
  fetchHtml: (url: string) => Promise<string | null>,
  extractFn: (html: string) => T,
  compareFn: (a: T, b: T) => string[],
  options?: VerifyOptions
): Promise<VerifiedScrapeResult<T>> {
  const maxAttempts = options?.maxAttempts ?? 2;
  const delayMs = options?.delayMs ?? 500;

  let previous: T | null = null;
  let lastDiffs: string[] = [];

  for (let attempt = 1; attempt <= maxAttempts; attempt++) {
    const html = await fetchHtml(url);
    if (!html) {
      lastDiffs.push(`Attempt ${attempt}: no HTML (fetch failed)`);
    } else {
      const data = extractFn(html);

      if (!previous) {
        previous = data;
      } else {
        const diffs = compareFn(previous, data);
        if (diffs.length === 0) {
          // OK: 2 extractions identiques
          return { ok: true, data, attempts: attempt };
        } else {
          lastDiffs = diffs;
        }
      }
    }

    if (attempt < maxAttempts) {
      await new Promise((res) => setTimeout(res, delayMs));
    }
  }

  return { ok: false, attempts: maxAttempts, diffs: lastDiffs };
}

fetchHtml = ton fetchWithRetry existant
extractFn = spécifique au cas (classement Serie A, liste produits, etc.)
compareFn = logique de comparaison pour ce type de données

import * as cheerio from 'cheerio';

function extractRankingTable(html: string): RankingRow[] {
  const $ = cheerio.load(html);
  const rows: RankingRow[] = [];

  // À adapter en fonction du site
  $('table tr').each((i, el) => {
    if (i === 0) return; // skip header
    const tds = $(el).find('td');
    if (tds.length < 2) return;

    const safeInt = (text: string) =>
      parseInt(text.replace(/[^\d\-]/g, ''), 10) || 0;

    const posText = $(tds[0]).text().trim();
    const nameText = $(tds[1]).text().trim();

    if (!nameText) return;

    // Ici tu peux mapper chaque colonne selon le site
    const row: RankingRow = {
      position: safeInt(posText),
      name: nameText,
      points: tds[tds.length - 1]
        ? safeInt($(tds[tds.length - 1]).text())
        : undefined,
      extra: {},
    };

    rows.push(row);
  });

  return rows;
}
Comparateur générique pour tableaux de classement
function compareRankingResults(a: RankingRow[], b: RankingRow[]): string[] {
  const diffs: string[] = [];

  if (a.length !== b.length) {
    diffs.push(`Row count mismatch: ${a.length} vs ${b.length}`);
  }

  const len = Math.min(a.length, b.length);
  for (let i = 0; i < len; i++) {
    const ra = a[i];
    const rb = b[i];

    if (ra.position !== rb.position) {
      diffs.push(`Row ${i + 1}: position ${ra.position} != ${rb.position}`);
    }
    if (ra.name !== rb.name) {
      diffs.push(
        `Row ${i + 1}: name "${ra.name}" != "${rb.name}"`,
      );
    }
    if (ra.points !== rb.points) {
      diffs.push(
        `Row ${i + 1}: points ${ra.points ?? 'n/a'} != ${rb.points ?? 'n/a'}`,
      );
    }
  }

  return diffs;
}

Comparateur de deux classements
export function compareRankingResults(
  a: RankingRow[],
  b: RankingRow[]
): string[] {
  const diffs: string[] = [];

  if (a.length !== b.length) {
    diffs.push(`Row count mismatch: ${a.length} vs ${b.length}`);
  }

  const len = Math.min(a.length, b.length);
  for (let i = 0; i < len; i++) {
    const ra = a[i];
    const rb = b[i];

    const fields: (keyof RankingRow)[] = [
      'position',
      'name',
      'points',
      'played',
      'wins',
      'draws',
      'losses',
      'goalsFor',
      'goalsAgainst',
    ];

    for (const f of fields) {
      if (ra[f] !== rb[f]) {
        diffs.push(
          `Row ${i + 1} (${ra.name} vs ${rb.name}) field ${String(
            f,
          )}: ${ra[f]} != ${rb[f]}`,
        );
      }
    }
  }

  return diffs;
}

export function extractArticle(html: string): ArticleData {
  const $ = cheerio.load(html);

  const title =
    $('meta[property="og:title"]').attr('content') ||
    $('h1').first().text().trim() ||
    $('title').text().trim() ||
    '';

  const body = $('body')
    .clone()
    .find('script, style, noscript, nav, footer, header')
    .remove()
    .end()
    .text()
    .replace(/\s+/g, ' ')
    .trim();

  const date =
    $('time[datetime]').attr('datetime') ||
    $('meta[property="article:published_time"]').attr('content') ||
    undefined;

  return { title, body, date };
}
