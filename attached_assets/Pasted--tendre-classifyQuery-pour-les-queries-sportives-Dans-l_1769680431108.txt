// Exemple d’extension de classifyQuery

export function classifyQuery(query: string): string {
  const lower = query.toLowerCase();

  // --- SPORTS (foot + autres) ---
  const footballKeywords = [
    "ligue des champions", "champions league", "ldc",
    "ligue 1", "ligue1", "l1",
    "premier league", "pl",
    "la liga", "liga",
    "serie a", "bundesliga",
    "score", "résultat", "resultat", "classement", "calendrier", "prochaine journée",
    "match", "buteur", "but", "carton rouge",
    "om", "marseille", "psg", "paris saint germain",
    "real madrid", "barça", "barcelona", "bayern", "man city", "manchester city"
  ];

  const basketballKeywords = [
    "nba", "euroleague", "euroleague", "basket", "basketball",
    "playoffs nba", "regular season", "score nba"
  ];

  const f1Keywords = [
    "formule 1", "f1", "grand prix", "grand prix f1", "classement pilotes", "classement constructeurs"
  ];

  // Football d’abord car c’est ton sport principal
  if (footballKeywords.some(k => lower.includes(k))) {
    return "sports_football";
  }

  if (basketballKeywords.some(k => lower.includes(k))) {
    return "sports_basketball";
  }

  if (f1Keywords.some(k => lower.includes(k))) {
    return "sports_motorsport";
  }

  // --- Reste de ta logique existante ---
  // (exemple générique, à adapter avec ce que tu as déjà)
  if (/* ton test conversational */ false) {
    return "conversational";
  }

  if (/* test news/temporal */ false) {
    return "news"; // ou "temporal"
  }

  // etc.

  return "factual";
}
