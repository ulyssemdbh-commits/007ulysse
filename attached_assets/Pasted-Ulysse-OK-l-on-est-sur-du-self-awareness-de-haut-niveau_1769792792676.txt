  mutationFn: async () => {
    const res = await fetch("/api/diagnostics/run");
    return res.json();
  },

    refetchHealth();
    refetchIssues();
    ...
  },
});
/api/diagnostics/run semble utiliser runDiagnostics() (la version light), pas runComprehensiveDiagnostic.
