import { chromium } from "playwright";
import * as cheerio from "cheerio";

const URL = "https://www.enligne.parionssport.fdj.fr/paris-football/coupes-d-europe/championsleague";

interface MatchOdds {
  homeTeam: string;
  awayTeam: string;
  odds1: number | null;
  oddsN: number | null;
  odds2: number | null;
}

function toNum(text: string | undefined): number | null {
  if (!text) return null;
  const n = Number(text.replace(",", ".").trim());
  return isNaN(n) ? null : n;
}

async function scrapePage(url: string): Promise<string> {
  const browser = await chromium.launch({ headless: true });
  const context = await browser.newContext({
    viewport: { width: 1366, height: 768 },
    javaScriptEnabled: true,
    locale: "fr-FR",
    userAgent:
      "Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/121.0 Safari/537.36"
  });

  const page = await context.newPage();
  const resp = await page.goto(url, { waitUntil: "networkidle", timeout: 45000 });

  if (!resp) {
    throw new Error("No response from page");
  }

  // laisse le temps au JS de charger les matchs
  await page.waitForTimeout(5000);

  const html = await page.content();

  await context.close();
  await browser.close();

  return html;
}

function parseChampionsLeague(html: string): MatchOdds[] {
  const $ = cheerio.load(html);
  const results: MatchOdds[] = [];

  // 👉 ICI : tu dois adapter aux vraies classes/data-attrs de FDJ
  // Inspecte un "match" dans le DOM avec F12, et ajuste selectors.
  const rows = $("[data-test='event-row'], .event-row, .c-events__item");

  rows.each((_, el) => {
    const row = $(el);

    const homeTeam =
      row
        .find("[data-test='event-home-team-name'], .event-row__participant--home, .c-events__team--home")
        .text()
        .trim();
    const awayTeam =
      row
        .find("[data-test='event-away-team-name'], .event-row__participant--away, .c-events__team--away")
        .text()
        .trim();

    if (!homeTeam || !awayTeam) return;

    const odd1Text =
      row
        .find("[data-selection='1'], [data-outcome='1'], .odd-1, [data-test='market-outcome-1']")
        .first()
        .text();
    const oddNText =
      row
        .find("[data-selection='N'], [data-outcome='N'], .odd-n, [data-test='market-outcome-N']")
        .first()
        .text();
    const odd2Text =
      row
        .find("[data-selection='2'], [data-outcome='2'], .odd-2, [data-test='market-outcome-2']")
        .first()
        .text();

    const odds1 = toNum(odd1Text);
    const oddsN = toNum(oddNText);
    const odds2 = toNum(odd2Text);

    results.push({
      homeTeam,
      awayTeam,
      odds1,
      oddsN,
      odds2
    });
  });

  return results;
}

async function main() {
  console.log("Scraping:", URL);
  const html = await scrapePage(URL);

  const matches = parseChampionsLeague(html);

  console.log(`Found ${matches.length} matches`);
  console.log(JSON.stringify(matches, null, 2));
}

main().catch(err => {
  console.error("ERROR:", err);
  process.exit(1);
});
