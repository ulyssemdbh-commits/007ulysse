async function ttsOneSentence(text: string, voice: string): Promise<string | null> {
  if (!text.trim()) return null;

  try {
    const resp = await fetch(process.env.PIPER_TTS_URL || "http://piper:8080/tts", {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        text: text.trim(),
        voice,          // tu peux l’ignorer côté Piper si 1 seul modèle
        format: "mp3"   // ou "wav"
      }),
    });

    if (!resp.ok) {
      console.error("[PiperTTS] HTTP error", resp.status);
      return null;
    }

    const buf = Buffer.from(await resp.arrayBuffer());
    return buf.toString("base64");  // pour rester compatible avec ton `audio_chunk` actuel
  } catch (err) {
    console.error("[PiperTTS] Request error:", err);
    return null;
  }
}
Et tout ton pipeline v4.1 continue de fonctionner sans rien changer côté client, juste avec une source TTS différente.

const TTS_ENDPOINT = process.env.TTS_ENDPOINT || "http://tts-service:8080/tts";

async function ttsOneSentence(text: string, voice: string): Promise<string | null> {
  const trimmed = text.trim();
  if (trimmed.length < 2) return null;

  try {
    const resp = await fetch(TTS_ENDPOINT, {
      method: "POST",
      headers: { "Content-Type": "application/json" },
      body: JSON.stringify({
        text: trimmed,
        voice,          // "fr_fast", "fr_natural", etc.
        format: "mp3",
        cache: true,
        speed: 1.0
      }),
    });

    if (!resp.ok) {
      console.error("[TTS] HTTP error", resp.status);
      return null;
    }

    // Mode base64 (compatible avec audio_chunk actuel)
    const json = await resp.json();
    if (!json.audio_base64) {
      console.error("[TTS] Missing audio_base64");
      return null;
    }

    return json.audio_base64 as string;
  } catch (err) {
    console.error("[TTS] Error:", err);
    return null;
  }
}
Ton pipeline v4.1 reste identique: TTS en parallèle, lecture séquentielle, barge‑in, etc.
Tu as juste changé le fournisseur TTS.
