4. Serveur Fastify + routes
src/server.ts
import Fastify from 'fastify';
import cors from '@fastify/cors';
import formBody from '@fastify/formbody';
import rateLimit from '@fastify/rate-limit';
import { registerRoutes } from './routes';
import { logger } from './utils/logger';
import { config } from './utils/config';

async function main() {
  const fastify = Fastify({
    logger,
  });

  await fastify.register(cors, { origin: true });
  await fastify.register(formBody);
  await fastify.register(rateLimit, {
    max: 1000,
    timeWindow: '1 minute',
  });

  registerRoutes(fastify);

  const port = config.PORT ?? 4000;
  const host = '0.0.0.0';

  try {
    await fastify.listen({ port, host });
    fastify.log.info(`Core scraper API listening on http://${host}:${port}`);
  } catch (err) {
    fastify.log.error(err);
    process.exit(1);
  }
}

main();

src/core/domainProfile.ts
import { RenderStrategy } from './types';

export interface DomainProfile {
  domain: string;
  defaultStrategy: RenderStrategy;
  jsRequired: boolean;
  rateLimitPerMinute: number;
  userAgents: string[];
  extractor?: string;
  lastSuccessStrategy?: RenderStrategy;
  lastUpdatedAt: string;
}

const defaultUA =
  'Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 ' +
  '(KHTML, like Gecko) Chrome/123.0.0.0 Safari/537.36';

export function defaultProfileFor(domain: string): DomainProfile {
  return {
    domain,
    defaultStrategy: 'http',
    jsRequired: false,
    rateLimitPerMinute: 300,
    userAgents: [defaultUA],
    lastUpdatedAt: new Date().toISOString(),
  };
}

// ⚠️ Version in-memory pour le squelette.
// En prod: remplace par Postgres / Redis.
const inMemoryProfiles = new Map<string, DomainProfile>();

export async function loadDomainProfile(url: string): Promise<DomainProfile> {
  const domain = new URL(url).hostname;
  const existing = inMemoryProfiles.get(domain);
  if (existing) return existing;
  const profile = defaultProfileFor(domain);
  inMemoryProfiles.set(domain, profile);
  return profile;
}

export async function updateDomainProfile(
  url: string,
  patch: Partial<DomainProfile>,
): Promise<void> {
  const domain = new URL(url).hostname;
  const current = await loadDomainProfile(url);
  const updated: DomainProfile = {
    ...current,
    ...patch,
    lastUpdatedAt: new Date().toISOString(),
  };
  inMemoryProfiles.set(domain, updated);
}
