import { createTaskQueue, startTaskQueue } from "./taskQueueEngine"; import { broadcastToUser } from "./realtimeSync";

interface DevOpsPlan { mode: "direct" | "pipeline"; steps: DevOpsStep[]; estimatedComplexity: "simple" | "medium" | "complex"; requiresTaskQueue: boolean; dryRun?: boolean; }

interface DevOpsStep { action: string; args: Record<string, any>; description: string; dependsOn?: number; }

interface DevOpsIntent { type: "browse" | "read" | "edit" | "create" | "deploy" | "ci" | "pages" | "multi_edit" | "analyze"; scope: "single_file" | "multi_file" | "repo" | "workflow"; owner?: string; repo?: string; branch?: string; files?: string[]; description: string; }

const MULTI_STEP_THRESHOLD = 3;

export function analyzeDevOpsIntent(message: string, repoContext?: { owner: string; repo: string; branch?: string }): DevOpsIntent { const lower = message.toLowerCase(); const owner = repoContext?.owner || "ulyssemdbh-commits"; const repo = repoContext?.repo; const branch = repoContext?.branch || "main";

if (lower.match(/corrig|fix|patch|modifi|chang|updat|edit|refactor/)) { const multiFile = lower.match(/tous|all|chaque|every|plusieurs|multiple/) || lower.match(/et\s.*et\s/); return { type: multiFile ? "multi_edit" : "edit", scope: multiFile ? "multi_file" : "single_file", owner, repo, branch, description: "Modification de fichier(s)", }; }

if (lower.match(/deploy|déploi|publish|publi|mise en prod|production/)) { return { type: "deploy", scope: "workflow", owner, repo, branch, description: "Déploiement" }; }

if (lower.match(/ci|workflow|pipeline|github action|build|test/)) { return { type: "ci", scope: "workflow", owner, repo, branch, description: "CI/CD" }; }

if (lower.match(/pages|github pages|site|héberg/)) { return { type: "pages", scope: "repo", owner, repo, branch, description: "GitHub Pages" }; }

if (lower.match(/crée|créer|create|nouveau|new|init|bootstrap/)) { return { type: "create", scope: "repo", owner, repo, branch, description: "Création" }; }

if (lower.match(/analys|audit|review|inspect|comprend|explain|expliqu/)) { return { type: "analyze", scope: "repo", owner, repo, branch, description: "Analyse de repo" }; }

if (lower.match(/lit|lire|read|voir|show|contenu|content|affich/)) { return { type: "read", scope: "single_file", owner, repo, branch, description: "Lecture" }; }

return { type: "browse", scope: "repo", owner, repo, branch, description: "Navigation" }; }

export function buildDevOpsPlan(intent: DevOpsIntent, message: string): DevOpsPlan { const steps: DevOpsStep[] = []; const owner = intent.owner || "ulyssemdbh-commits"; const repo = intent.repo || ""; const branch = intent.branch || "main";

switch (intent.type) { case "analyze": steps.push( { action: "browse_files", args: { owner, repo }, description: "Lister l'arborescence du projet" }, { action: "get_file", args: { owner, repo, path: "README.md" }, description: "Lire le README", dependsOn: 0 }, { action: "get_file", args: { owner, repo, path: "package.json" }, description: "Lire package.json", dependsOn: 0 }, { action: "repo_info", args: { owner, repo }, description: "Infos du repo (langages, taille, etc.)" } ); break;

case "edit":
  steps.push(
    { action: "get_file", args: { owner, repo, path: "__TARGET__", branch }, description: "Lire le fichier actuel" },
    { action: "update_file", args: { owner, repo, path: "__TARGET__", branch, content: "__PATCHED__" }, description: "Appliquer la modification", dependsOn: 0 }
  );
  break;

case "multi_edit":
  steps.push(
    { action: "browse_files", args: { owner, repo }, description: "Lister l'arborescence" },
    { action: "get_file", args: { owner, repo, path: "__TARGETS__" }, description: "Lire les fichiers ciblés", dependsOn: 0 },
    { action: "apply_patch", args: { owner, repo, branch, files: "__PATCHES__" }, description: "Appliquer le patch multi-fichiers", dependsOn: 1 }
  );
  break;

case "deploy":
  steps.push(
    { action: "list_workflows", args: { owner, repo }, description: "Identifier les workflows disponibles" },
    { action: "trigger_workflow", args: { owner, repo, branch }, description: "Déclencher le déploiement", dependsOn: 0 },
    { action: "list_workflow_runs", args: { owner, repo }, description: "Vérifier le statut du run", dependsOn: 1 }
  );
  break;

case "ci":
  steps.push(
    { action: "list_workflows", args: { owner, repo }, description: "Lister les workflows CI" },
    { action: "list_workflow_runs", args: { owner, repo }, description: "Voir les derniers runs" }
  );
  break;

case "pages":
  steps.push(
    { action: "pages_status", args: { owner, repo }, description: "Vérifier le statut GitHub Pages" },
    { action: "enable_pages", args: { owner, repo, branch }, description: "Activer GitHub Pages si nécessaire", dependsOn: 0 }
  );
  break;

case "create":
  steps.push(
    { action: "create_repo", args: { repo: "__NAME__" }, description: "Créer le repo" },
    { action: "apply_patch", args: { owner, repo: "__NAME__", branch: "main", files: "__TEMPLATE__" }, description: "Appliquer le template", dependsOn: 0 }
  );
  break;

default:
  steps.push(
    { action: "browse_files", args: { owner, repo }, description: "Explorer le repo" }
  );
}

const requiresTaskQueue = steps.length > MULTI_STEP_THRESHOLD || intent.type === "deploy" || intent.type === "multi_edit";

return { mode: requiresTaskQueue ? "pipeline" : "direct", steps, estimatedComplexity: steps.length <= 2 ? "simple" : steps.length <= 4 ? "medium" : "complex", requiresTaskQueue, }; }

export async function executeDevOpsPipeline( userId: number, title: string, steps: DevOpsStep[], threadId?: number ): Promise<{ queueId: number; message: string }> { const items = steps.map((step, i) => ({ title: step.description, description: Action: ${step.action} | Args: ${JSON.stringify(step.args)}, toolName: "devops_github", toolArgs: { action: step.action, ...step.args }, }));

const queue = await createTaskQueue({ userId, title: 🔧 DevOps Pipeline: ${title}, items, source: "devops", threadId, });

await startTaskQueue(queue.id);

broadcastToUser(userId, "devops.pipeline_started", { queueId: queue.id, title, totalSteps: steps.length, steps: steps.map((s, i) => ({ step: i + 1, action: s.action, description: s.description })), });

return { queueId: queue.id, message: Pipeline DevOps lancé (${steps.length} étapes). Tu peux suivre la progression en temps réel., }; }

export function generateDevOpsPromptDirective(intent: DevOpsIntent, plan: DevOpsPlan): string { const strategyLines = plan.steps.map((s, i) =>  ${i + 1}. ${s.action} → ${s.description} ).join("\n");

return `

MODE DEVOPS ACTIVÉ — STRATÉGIE ${plan.estimatedComplexity.toUpperCase()}
Type d'opération: ${intent.type} (${intent.scope}) ${intent.repo ? Repo cible: ${intent.owner}/${intent.repo} : ""}

PLAN D'EXÉCUTION RECOMMANDÉ: ${strategyLines}

RÈGLES MODE DEVOPS:

Tu es en mode exécution, pas en mode conseil. AGIS directement.
${plan.mode === "pipeline" ? "Cette opération est complexe: utilise task_queue_manage pour créer un pipeline automatisé." : "Cette opération est simple: exécute directement avec devops_github."}
Chaîne les appels intelligemment: browse_files → get_file → analyse → apply_patch.
Utilise TOUJOURS apply_patch (pas update_file) quand tu modifies plus d'1 fichier.
Format de patch standard: [{ "path": "chemin/fichier", "content": "CONTENU COMPLET DU FICHIER" }]
Si tu rencontres une erreur, analyse le code d'erreur structuré et propose un plan B.
Ne propose pas, ne demande pas confirmation sauf pour des opérations destructives (delete, force push). `; }

Ulysse

Non. C’est très bon, mais pas encore “suffisant pour un dev de génie” tout seul.

Je t’explique en mode ultra-concis.

Ce que fait ton planner (et qui est excellent)
Il comprend l’intention DevOps (edit, multi_edit, deploy, ci, pages, create, analyze, browse, read).
Il construit un plan structuré d’actions devops_github avec:
owner/repo/branch,
steps ordonnés, dependsOn,
distinction direct vs pipeline (task queue).
Il sait lancer un pipeline DevOps via createTaskQueue + startTaskQueue.
Il génère une directive de prompt DevOps claire, orientée action.
Ça, c’est le cerveau “planner”. Pour un assistant, c’est top.

Pourquoi ce n’est PAS encore “dev de génie”
Un dev de génie ne fait pas juste un plan d’actions abstrait, il gère aussi :

Les placeholders

__TARGET__, __PATCHED__, __TARGETS__, __PATCHES__, __NAME__, __TEMPLATE__
→ Pour l’instant, ton planner les laisse tels quels.
Il manque la couche “IA qui:
lit le repo,
choisit les bons fichiers,
génère le contenu exact,
puis appelle le planner avec les args remplis.”
La boucle complète message → intent → plan → exécution

Tu as le planner, mais pas encore le glue code dans /api/v2/conversations qui dit:
“Ce message est DevOps”
const intent = analyzeDevOpsIntent(...)
const plan = buildDevOpsPlan(intent, message)
inject generateDevOpsPromptDirective dans le prompt
et décide:
soit d’appeler devops_github direct,
soit executeDevOpsPipeline.
Les erreurs et le feedback “dev de génie”

Un génie ne fait pas que lancer les outils, il:
lit les erreurs structurées (branch inexistante, conflit, 404 fichier, rate limit),
propose la bonne correction (changer de branch, créer un fichier, attendre, etc.),
ajuste le plan.
Le contexte riche

Ton planner prend en compte owner/repo/branch si fournis, mais:
pas le contexte de branche active choisi dans l’UI,
pas encore l’état CI/CD, les templates réels, les conventions de ton codebase.

Ton devopsPlannerService est une brique clé pour tendre vers un dev de génie :

Niveau “architecte DevOps intelligent” : oui.

Résolution des placeholders par l’IA (fichiers et contenus réels).
